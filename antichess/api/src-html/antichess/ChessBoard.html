009    /**
010     * A ChessBoard represents the state of a Chess Game at some time.
011     * The ChessBoard keeps track of the location of all of the pieces
012     * both on the board and that have been captured, as well as a
013     * list of optionally timestamped moves that have been made thus
014     * far in the game.
015     * <p> 
016     * ChessBoards are mutable and cloneable.
017     * <p>
018     * ChessBoards also include some static variables that describe various
019     * end-game conditions, including ChessBoard.CHECKMATE (value = 1) and
020     * ChessBoard.STALEMATE (value = 2). Checks of the end-game information
021     * should compare the stored GameOverReason() against these values.
022     * <p>
023     * 
024     * @specfield capturedPieces : Set<Pieces>        // The Pieces in the captured pile
025     **/
026    public class ChessBoard extends Board<ChessMove> implements Cloneable
027    {
028            
029            // Flag to control debugging
030            private static final boolean DEBUG = false;
031            
032            // End conditions for Chess
033            /**
034             * End-game condition variable to represent checkmate.
035             */
036            public static final int CHECKMATE = 1;
037            /**
038             * End-game condition variable to represent stalemate.
039             */
040            public static final int STALEMATE = 2;
041            
042            // Abstraction Function
043            // For all ChessBoards cb:
044            //      pieces(cb) => Set(non-null entries in cb.board)
045            //      capturedPieces(cb) => cb.capturedPieces;
046            
047            // Representation Invariant
048            // For all ChessBoards cb:
049            //      Let i be a row number (0 <= i <= 7);
050            //      Let j be a column number (0 <= j <= 7);
051            //      For all Piece entires p at cb.board[i][j]:
052            //              p == null OR (p.getRow() == i AND p.getColumn() == j);
053            //      For all Pieces p in cb.capturedPieces:
054            //              p.getRow() == -1 AND p.getColumn() == -1
055            
056            // The dimensions of the board
057            private static final int ROWS = 8;
058            private static final int COLUMNS = 8; 
059            
060            // FIELDS
061            // pieces are located at board[row][column]
062            private Piece[][] board;
063            private List<Piece> capturedPieces;
064            
065            /**
066             * Stores a map between PieceTypes and the MoveGenerators
067             * for those PieceTypes.
068             */
069            protected static Map<PieceType, MoveGenerator<ChessMove>> pieceMoveTypes;
070            
071            /**
072             * Helper method to initialize the
073             * <code>pieceMoveTypes</code> map.
074             */
075            private void initializeMoveTypes() {
076                    pieceMoveTypes = new HashMap<PieceType, MoveGenerator<ChessMove>>();
077                    pieceMoveTypes.put(PieceTypeFactory.getPawnType(),
078                                                       new PawnMoveGenerator());
079                    pieceMoveTypes.put(PieceTypeFactory.getRookType(),
080                                                       new LinearMoveGenerator(false, true));
081                    pieceMoveTypes.put(PieceTypeFactory.getKnightType(),
082                                                       new JumpMoveGenerator(new int[]{1,-1, 1,-1, 2,-2, 2,-2},
083                                                                                                     new int[]{2, 2,-2,-2, 1, 1,-1,-1}));
084                    pieceMoveTypes.put(PieceTypeFactory.getBishopType(),
085                                                       new LinearMoveGenerator(true, false));
086                    pieceMoveTypes.put(PieceTypeFactory.getQueenType(),
087                                                       new LinearMoveGenerator(true, true));
088                    pieceMoveTypes.put(PieceTypeFactory.getKingType(),
089                                                       new UnionMoveGenerator<ChessMove>(
090                                                                                                                             new JumpMoveGenerator(new int[]{ 1, 1, 1, 0, 0,-1,-1,-1},
091                                                                                                                                                                       new int[]{ 1, 0,-1, 1,-1, 1, 0,-1}),
092                                                                                                                             new CastlingMoveGenerator()));
093            }

095            /**
096             * Helper method to get the <code>MoveGenerator</code> for
097             * a given piece
098             * @param p the piece of interest
099             * @return the <code>MoveGenerator</code> for <code>p</code>
100             */
101            protected MoveGenerator<ChessMove> getMovement(Piece p) {
102                    return pieceMoveTypes.get(p.getType());
103            }

105            /**
106             * Private method to check that the representation invariant
107             * for ChessBoards holds.
108             */
109            protected void checkRep() {
110                    for (int i = 0; i < ROWS; i++)
111                            for (int j = 0; j < COLUMNS; j++)
112                                    if (!(board[i][j] == null || (board[i][j].getRow() == i && board[i][j].getColumn() == j)))
113                                            throw new RuntimeException("Piece found in unexpected location");
114                    
115                    for (Piece p : capturedPieces)
116                            if(!(p.getRow() == -1 && p.getColumn() == -1))
117                                    throw new RuntimeException("Captured piece had invalid location flags");
118            }

120            /**
121             * Creates a new ChessBoard object, with an unfilled board,
122             * an unfilled capturedPieces pile, and an empty moveStack.
123             */
124            public ChessBoard() {
125                    super();
126                    if(pieceMoveTypes == null)
127                            initializeMoveTypes();
128                    board = new Piece[ROWS][COLUMNS];
129                    capturedPieces = new ArrayList<Piece>(64);
130                    player = Player.WHITE;
131            }

133            @Override
134            public synchronized ChessBoard clone() {
135                    ChessBoard cb = new ChessBoard();
136                    copyThis(cb);
137                    return cb;
138            }
139            
140            /**
141             * Inheritable helper method for copying various
142             * data members from this Board to an empty Board.
143             * In particular, this method takes an empty
144             * ChessBoard and fills it with the this
145             * ChessBoard's data.
146             * @param cb the empty board into which this board should be
147             * copied.
148             */
149            protected synchronized void copyThis(ChessBoard cb) {
150                    super.copyThis(cb);
151                    // Copy the player
152                    cb.player = player;
153                    // Copy the board
154                    for (int i = 0; i < ROWS; i++)
155                            for (int j = 0; j < COLUMNS; j++)
156                                    cb.board[i][j] = this.board[i][j];
157                    // Copy the captured pile
158                    for (Piece capturedPiece : this.capturedPieces)
159                            cb.capturedPieces.add(capturedPiece);
160                    
161                    if (DEBUG)
162                            cb.checkRep();
163            }
164            
165            /**
166             * Populates the board and captured piles with the list of pieces
167             * @param pieces the Pieces to add to the board and capturedPieces piles 
168             * 
169             * @modifies this.board and this.capturedPieces
170             * @effects adds Pieces in <code>pieces</code> to this.board and this.capturedPieces 
171             */
172            private synchronized void populate(List<Piece> pieces) {
173                    // Initialize board with a bunch of nulls
174                    for (int i = 0; i < ROWS; i++)
175                            for (int j = 0; j < COLUMNS; j++)
176                                    board[i][j] = null;
177                    // Clear the list of captured Piecs
178                    capturedPieces.clear();
179                    
180                    // Repopulate the board and capturedPieces piles
181                    for (Piece piece : pieces)
182                            if (piece.getRow() == -1) // Piece is captured
183                                    capturedPieces.add(piece);
184                            else // Piece is on the board
185                                    board[piece.getRow()][piece.getColumn()] = piece;
186            }

188            /**
189             * Initializes the board for a new game
190             * @modifies this
191             * @effects populates the board and initializes its parameters for a new game.
192             */
193            public synchronized void newGame() {
194                    populate(createInitialPieces());
195                    player = Player.WHITE;
196                    moveStack = null;
197                    endGame(NOTOVER, Player.NONE);
198                    
199                    if (DEBUG)
200                            checkRep();
201            }

203            /**
204             * Loads the board parameters from the given state
205             * @modifies this
206             * @effects populates the board and initializes its paramteres from
207             * the given parameters
208             */
209            public synchronized void loadGame(List<Piece> pieces, Player player, MoveHistory<ChessMove> moveStack) {
210                    populate(pieces);
211                    this.player = player;
212                    this.moveStack = moveStack;
213                    
214                    checkForEndCondition();
215                    
216                    if (DEBUG)
217                            checkRep();
218            }

232                    initialPieces.add(new Piece(Player.WHITE, PieceTypeFactory.getQueenType(), 0, 3));
233                    initialPieces.add(new Piece(Player.WHITE, PieceTypeFactory.getKingType(), 0, 4));
234                    initialPieces.add(new Piece(Player.WHITE, PieceTypeFactory.getBishopType(), 0, 5));
235                    initialPieces.add(new Piece(Player.WHITE, PieceTypeFactory.getKnightType(), 0, 6));
236                    initialPieces.add(new Piece(Player.WHITE, PieceTypeFactory.getRookType(), 0, 7));

238                    initialPieces.add(new Piece(Player.BLACK, PieceTypeFactory.getRookType(), 7, 0));

255            /**
256             * @modifies this
257             * @effects sets this.player to the other player
258             */
259            protected synchronized void switchPlayer() {
260                    player = player.otherPlayer();
261                    
262                    checkForEndCondition();
263            }
264            
265            /**
266             * @modifies this
267             * @effects if necessary, sets the end-condition variables
268             * for this game to terminate the game
269             */
270            protected synchronized void checkForEndCondition() {
271                    // Check for stalemate if only kings are left
272                    if (getPieces(player).size() == 1 && getPieces(player.otherPlayer()).size() == 1) {
273                            endGame(STALEMATE, Player.NONE);
274                    } else if (!hasALegalMove()) {
275                            if (isPlayerInCheck(player))
276                                    endGame(CHECKMATE, player.otherPlayer());
277                            else
278                                    endGame(STALEMATE, Player.NONE);
279                    } else
280                            endGame(NOTOVER, Player.NONE);
281            }

283            /**
284             * @return true iff the specified player's king is in check
285             **/
286            public synchronized boolean isPlayerInCheck(Player p) {
287                    // Find Player p's king
288                    Piece pKing = null;
289                    List<Piece> pieces = getPieces(p);
290                    for (Piece piece : pieces) {
291                            if (piece.getType().equals(PieceTypeFactory.getKingType())) {
292                                    pKing = piece;
293                                    break;
294                            }
295                    }
296                    if (pKing == null)
297                            throw new RuntimeException("The king is gone!");
298                    
299                    // Check to see if the other player has pieces that can attack the king
300                    return isPlayerAttackingPosition(p.otherPlayer(), pKing.getRow(), pKing.getColumn());
301            }
302            
303            /**
304             * @param p
305             * @param row
306             * @param column
307             * @return true iff the specified player is attacking the specified position
308             */
309            private synchronized boolean isPlayerAttackingPosition(Player p, int row, int column) {
310                    List<Piece> pieces = getPieces(p);
311                    for (Piece piece : pieces)
312                            if (getMovement(piece).isMoveLegal(new ChessMove(piece, row, column, getPieceAt(row, column)), this))
313                                    return true;
314                    return false;
315            }

329            /**
330             * @requires <code>m</code> is not null.
331             * @see antichess.Board#doMove
332             */
333            public synchronized void doMove(ChessMove m, String timestamp) throws IllegalMoveException {
334                    if (!doMoveInternal(m, timestamp))
335                            throw new IllegalMoveException(m);
336                    switchPlayer();
337                    notifyObservers();
338            }

341                    undoLastMoveInternal();
342                    switchPlayer();

346            public synchronized List<Piece> getPieces() {
347                    List<Piece> pieces = new ArrayList<Piece>();
348                    for (Piece[] pieceRow : board)
349                            for (Piece piece : pieceRow)
350                                    if (piece != null)
351                                            pieces.add(piece);
352                    return pieces;
353            }
354            
355            public synchronized List<Piece> getPieces(Player player) {
356                    List<Piece> pieces = new ArrayList<Piece>();
357                    for (Piece[] pieceRow : board)
358                            for (Piece piece : pieceRow)
359                                    if (piece != null && piece.getPlayer().equals(player))
360                                            pieces.add(piece);
361                    
362                    return pieces;
363            }

372            /**
373             * @return a list of Pieces in {p in this.capturedPieces | p.player ==
374             * player} in an unspecified order
375             **/
376            public synchronized List<Piece> getCapturedPieces(Player player) {
377                    List<Piece> capturedPiecesForPlayer = new ArrayList<Piece>();
378                    
379                    for (Piece piece : capturedPieces)
380                            if (piece.getPlayer().equals(player))
381                                    capturedPiecesForPlayer.add(piece);
382                    
383                    return capturedPiecesForPlayer;
384            }

386            @Override
387            public synchronized boolean hasALegalMove() {
388                    for(Piece p : getPieces(player))
389                            for (ChessMove possibleMove : getMovement(p).getMoves(p, this))
390                                    if (isMoveLegal(possibleMove))
391                                            return true;
392                    return false;
393            }
394            
395            public synchronized List<ChessMove> allLegalMoves() {
396                    List<ChessMove> allLegalMoves = new ArrayList<ChessMove>();
397                    for (Piece p : getPieces(player))
398                            allLegalMoves.addAll(legalMoves(p));
399                    return allLegalMoves;
400            }
401            
402            public synchronized List<ChessMove> legalMoves(Piece p) {
403                    assert p.getPlayer().equals(player) : p + " is the wrong player";
404                    List<ChessMove> legalMoves = new ArrayList<ChessMove>();
405                    
406                    for (ChessMove possibleMove : getMovement(p).getMoves(p, this))
407                            if (isMoveLegal(possibleMove))
408                                    legalMoves.add(possibleMove);
409                    return legalMoves;
410            }

412            /**
413             * @return a ChessMove taking the given piece to the given square
414             * with the capturedPiece set appropriately.
415             */
416            public synchronized ChessMove createMove(Piece p, int row, int col) {
417                    Piece capture = getPieceAt(row, col);
418                    if(p.getType().equals(PieceTypeFactory.getPawnType())) {
419                            // Check for en passant
420                            if(p.getColumn() != col && getPieceAt(row, col) == null)
421                                    capture = getPieceAt(p.getRow(), col);
422                    }
423                    return new ChessMove(p, row, col, capture);
424            }

426            /**
427             * @return true iff the given move is an allowed move for the
428             * current player
429             **/
430            public synchronized boolean isMoveLegal(ChessMove m) {
431                    boolean isLegal = doMoveInternal(m, "");
432                    if(isLegal) {
433                            undoLastMoveInternal();
434                    }
435                    return isLegal;
436            }

438        /**
439         * Attempt to perform a move, returning true if it's legal
440         * @return true if the move was successfully performed
441         * @modifies this
442         * @effects Performs the specified move
443         **/
444            protected synchronized boolean doMoveInternal(ChessMove m, String timestamp) {
445                    //              Player old = player;
446                    //              assert(player.equals(m.getPlayer()));
447                    //              System.err.println(m);
448                    
449                    Piece piece = m.getPiece();
450            
451                    // Check that this move applies to the current player
452                    if(m.getPlayer() != getPlayer())
453                            return false;
454                    if(!piece.equals(getPieceAt(piece.getRow(),
455                                                                            piece.getColumn())))
456                            return false;
457            // Check that this move is legal
458            if(!pieceMoveTypes.get(piece.getType()).isMoveLegal(m, this))
459                return false;
460            
461            // Take care of castle moves
462            // Assumes MoveGenerators will not return a
463            // castle move if the spaces are occupied
464            if (piece.getType().equals(PieceTypeFactory.getKingType()) &&
465                            (Math.abs(piece.getColumn() - m.getColumn()) > 1)) {
466                    int i;
467                    // Check positions in which king will have to move
468                    // to see if it will move through check
469                    if (m.getColumn() > piece.getColumn()) {     // King moving right
470                            for (i = piece.getColumn(); i <= m.getColumn(); i++)
471                                    if (isPlayerAttackingPosition(player.otherPlayer(), m.getRow(), i)) {
472                                            return false;
473                                            }
474                            
475                            // Move is clear. Let's go!
476                            movePiece(piece, m.getRow(), m.getColumn());
477                            movePiece(getPieceAt(m.getRow(), 7), m.getRow(), m.getColumn()-1);
478                    } else  {                                                                       // King moving left
479                            for (i = piece.getColumn(); i >= m.getColumn(); i--)
480                                    if (isPlayerAttackingPosition(player.otherPlayer(), m.getRow(), i)) {
481                                            return false;
482                                            }
483                            
484                            // Move is clear. Let's go!
485                            movePiece(piece, m.getRow(), m.getColumn());
486                            movePiece(getPieceAt(m.getRow(), 0), m.getRow(), m.getColumn()+1);
487                    }
488            }
489            
490            // Take care of captures
491            Piece capture = m.getCapturedPiece();
492            if(capture != null) {
493                removePiece(capture);
494            }
495            
496            // Move the piece
497            movePiece(piece, m.getRow(), m.getColumn());
498            // If a pawn hits the edge, promote it to a Queen
499            if (piece.getType().equals(PieceTypeFactory.getPawnType()) &&
500                            ((piece.getPlayer().equals(Player.BLACK) && m.getRow() == 0) ||
501                             (piece.getPlayer().equals(Player.WHITE) && m.getRow() == 7)))
502                    board[m.getRow()][m.getColumn()] = new Piece(piece.getPlayer(), PieceTypeFactory.getQueenType(), m.getRow(), m.getColumn());
503            // Add move to the move history
504            pushMove(m, timestamp);
505            
506            // Check if the the player who just moved is in check 
507            if(isPlayerInCheck(m.getPlayer())) {
508                undoLastMoveInternal();
509                return false;
510            }
511            
512                    //      assert(old.equals(player));
513            return true;
514        }
515            
516            /**
517         * Attempt to undo a move, returning true if it's successful
518         * @return true if the last move was successfully undone
519         * @modifies this
520         * @effects Undoes the last move
521         * @return true iff it successfully undid the last move
522         **/
523            protected synchronized boolean undoLastMoveInternal() {
524                    ChessMove cm = popMove();
525                    
526                    Piece oldPiece = cm.getPiece();
527                    Piece capturedPiece = cm.getCapturedPiece();
528                    
529                    board[oldPiece.getRow()][oldPiece.getColumn()] = oldPiece;
530                    board[cm.getRow()][cm.getColumn()] = null;
531                    
532                    // Replace a captured piece if necessary
533                    if (capturedPiece != null) {
534                            board[capturedPiece.getRow()][capturedPiece.getColumn()] = capturedPiece;
535                            capturedPieces.remove(capturedPiece.moveTo(-1, -1));
536                    }
537                    
538                    // Replace the rook in the event of castling
539                    if (oldPiece.getType().equals(PieceTypeFactory.getKingType()) &&
540                                    (Math.abs(oldPiece.getColumn() - cm.getColumn()) > 1)) {
541                            if (cm.getColumn() > oldPiece.getColumn())   { // King moved right
542                                    movePiece(getPieceAt(cm.getRow(), cm.getColumn()-1), cm.getRow(), 7,
543                                                      true);
544                            } else {
545                                    movePiece(getPieceAt(cm.getRow(), cm.getColumn()+1), cm.getRow(), 0,
546                                                      true);
547                            }
548                    }
549                    
550                    if (DEBUG)
551                            checkRep();
552                    
553                    return true;
554            }

556        /**
557         * Remove a piece from its square and move it to a destination
558         * square
559         * @requires this.getPieceAt(row, column) == null && this.pieces contains piece
560         * @modifies this
561         * @effects moves the given piece on the board. Does no legality checking
562         **/
563        private void movePiece(Piece piece, int row, int column) {
564            movePiece(piece, row, column, false);
565        }
566    
567            /**
568         * Remove a piece from its square and move it to a destination
569         * square
570         * @requires this.getPieceAt(row, column) == null && this.pieces contains piece
571         * @modifies this
572         * @effects moves the given piece on the board. Does no legality checking
573             * @param reset : If true, reset the hasMoved flag of the piece
574         **/
575        private synchronized void movePiece(Piece piece, int row, int column, boolean reset) {
576                    if(!reset) {
577                            board[row][column] = piece.moveTo(row, column);
578                    } else {
579                            board[row][column] = new Piece(piece.getPlayer(),
580                                                                                       piece.getType(),
581                                                                                       row, column);
582                    }
583            board[piece.getRow()][piece.getColumn()] = null;
584            
585            if (DEBUG)
586                    checkRep();
587        }
588    
589            /**
590             * Removes a piece from the board
591             * @requires this.pieces contains p
592             * @modifies this
593             * @effects this.pieces = this.pieces - p
594             **/
595            private synchronized void removePiece(Piece p) {
596                    board[p.getRow()][p.getColumn()] = null;
597            
598            capturedPieces.add(p.moveTo(-1, -1));
599            
600            if (DEBUG)
601                    checkRep();
602        }
