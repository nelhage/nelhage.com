003    package antichess.net;
004    
005    import antichess.*;
006    import net.antichess.ai.ChatProxy;
007    import net.antichess.ai.ChatHistoryViewer;
008    import net.antichess.ai.AiPlayer;
009    import java.util.*;
010    
011    /**
012     * <code>AiPlayerFactory</code> is an implementation of the
013     * antichess.net <code>AiPlayerFactory</code> interface.
014     * 
015     **/
016    public class AiPlayerFactory implements net.antichess.ai.AiPlayerFactory
017    {
018            public final String SPRING_2007_RULESET = "6170-spring-2007-encastle";
019            public AiPlayer createPlayer(boolean isWhite,
020                                                      Properties gameProperties,
021                                                      ChatProxy chatProxy,
022                                                      ChatHistoryViewer chatHistoryViewer,
023                                                      byte[] persistentState,
024                                                      int persistentStateSizeLimit) {
025                    if(!gameProperties.get(PROPERTY_RULE_SET).equals(SPRING_2007_RULESET))
026                            return null;
027                    return new MachinePlayer(isWhite, 0);
028            }
029    
030            public Collection<String> getSupportedRulesets() {
031                    Collection<String> rulesets = new ArrayList<String>();
032                    rulesets.add(SPRING_2007_RULESET);
033                    return rulesets;
034            }
035    
036    }
