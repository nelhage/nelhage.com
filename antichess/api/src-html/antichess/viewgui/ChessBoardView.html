032    /**
033     * ChessBoardView is the graphic representation of a Board for use in the
034     * view.  ChessBoardView extends the abstract class BoardView. 
035     *
036     * The ChessBoardView draws the viewedBoard and allows the user to input 
037     * moves when requested.
038     * 
039     * When setAcceptingInput is called the Board processes the users mouse clicks
040     * and reports a click from a square containing a piece for the HumanPlayer specified
041     * in the parameters to setAcceptingInput to a destination square.  Determining if this
042     * is a valid move is the responsibility of that HumanPlayer.  
043     * setNotAcceptingInput cancels out a setAcceptingInput. 
044     * 
045     * 
046     * A ChessBoardView requires that imageMap contain the following images:
047     *              RAntiking
048     *                      RBishop
049     *                      RKing
050     *                      RKnight
051     *                      RPawn
052     *                      RQueen
053     *                      RRook
054     *                      WAntiking
055     *                      WBishop
056     *                      WKing
057     *                      WKnight
058     *                      WPawn
059     *                      WQueen
060     *                      WRook
061     *                      board
062     * 
063     * 
064     * @author nlharr
065     *
066     * @specfield acceptingInput    //boolean that determines which state we are in
067     * @specfield startLocation             //Stores the value of the start location on the board
068     *                                                              //for the user inputed move
069     * @specfield endLocation               //Stores the value of the end location on the board
070     *                                                              //for the user inputed move
071     * @specfield boardPlayer               //stores the human player the BoardView is updates information to
072     * @specfield viewedBoard               //the ChessBoard that this ChessGUI is displaying
073     * @specfield showPiecesWithMoves       //boolean
074     * @specfield showPossibleMoves //boolean that determines whether squares are drawn over possible moves
075     * @specfield showLastMoves             //boolean that determines whether a square highlighting the last move is shown
076     * @specfield discoMode                 //boolean storing if we are in disco mode
077     * @specfield imageMap                  //ImageMap
078     * @specfield showKitty                 //boolean that determines whether we show an image of a kitty
079     */

334            /**
335             * Private method that draws a colored square over the specified square in the graphics g
336             * The square is drawn wtih the Composite of SRC_OVER so the alpha values for the
337             * color are taken into account.
338             * 
339             * @requires board != null, g != null, color != null
340             *                       0<=row<#rows in board 
341             *                       0<=column<#columns in board
342             * @effects Draws a square over the square identified by row and column 
343             */
344            
345            private void paintSquare(int row, int column, Color color, Graphics g, ChessBoard board){
346                    Graphics2D g2 = (Graphics2D)g;
347                    int leftPixel = squareWidth*column+boardXOffset;
348                    int topPixel = (board.getRows()-1-row)*squareWidth+boardYOffset;
349                    g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER));
350                    g2.setPaint(color);
351                    g2.fill(new Rectangle2D.Double(leftPixel, topPixel,
352                    squareWidth,
353                    squareHeight));
354                    
355            }

357            /**
358             *  Private method for highlighting the legal moves available to a piece.
359             *  
360             *      
361             * @requires board != null, g != null, color != null
362             *                       0<=row<#rows in board 
363             *                       0<=column<#columns in board
364             * @effects Gets the piece at row, column.  If the piece has available moves, it
365             *      calls paintSquare over the destinations for all the possible moves. 
366             *  
367             */
368            private void paintPossibleMoves(int row, int column, Color color, Graphics g, ChessBoard board){
369                    Piece pieceAt = board.getPieceAt(row, column);
370                    List<ChessMove> availableMoves = board.legalMoves(pieceAt);
371                    for(ChessMove tempMove : availableMoves){
372                            paintSquare(tempMove.getRow(), tempMove.getColumn(), color, g, board);
373                    }
374                    
375            }

378             *  Private method for highlighting all the pieces that can make legal moves

381             * @effects finds all the pieces that have legal moves and then calls
382             * paintSquare over those squares
383             */
384            private void paintHasMoves(Color color, Graphics g, ChessBoard board){
385                    for (Piece pc : board.getPieces()){
386                            if (board.legalMoves(pc).size()>0){
387                                    this.paintSquare(pc.getRow(), pc.getColumn(), color, g, board);
388                            }
389                    }

393            /**
394             * Private class in order to create a Comparator for pieces that sorts
395             * the pieces bottom up.  This is so when drawing the pieces occlusion
396             * happens properly.
397             * @author nlharr
398             *
399             */
400            private class PieceComparer implements Comparator<Piece>{
401                    /**
402                     * @returns     -1 if a.getRow()<b.getRow()
403                     *                      0 if a.getRow()==b.getRow()
404                     *                      -1 if a.getRow()>b.getRow()
405                     */
406                    public int compare(Piece a, Piece b){
407                                    if (a.getRow() > b.getRow()){
408                                            return -1;
409                                    }else if (a.getRow()==b.getRow()){
410                                            return 0;
411                                    }else {
412                                            return 1;
413                                    }               
414                    }
415            }

417            /**
418             *  Private method for drawing all the pieces
419             *  
420             * @requires board != null, g != null
421             * @effects sorts the pieces from bottom up and then draws them in
422             *                      their appropriate locations
423             */
424            private void paintPieces(Graphics g, ChessBoard board){
425                    List<Piece> pieceSet = board.getPieces();
426                  
427            //sorts the array
428            Piece[] sortedPieceArray =  pieceSet.toArray(new Piece[pieceSet.size()]);
429            Arrays.sort(sortedPieceArray, new PieceComparer());
430            
431            BufferedImage currentImage;
432            for (Piece currentPiece : sortedPieceArray){
433                    
434                    //gets the image appropriate for the Player
435                    if (currentPiece.getPlayer().equals(Player.WHITE) ){
436                            currentImage = imageMap.getImage(  "W" + currentPiece.getType().getName());
437                    }else{
438                            currentImage = imageMap.getImage(  "R" + currentPiece.getType().getName());
439                    }
440                  
441                    //draws the image
442                    g.drawImage(currentImage, 
443                                    getPiecePixelX(currentPiece.getColumn(), currentImage), 
444                                    getRowPixel(currentPiece.getRow(), currentImage), null
445                                    );   
446                }
447            }

449            /**
450             *  Private method for highlighting the beginning and ending destination of the
451             *  last move
452             *  
453             * @requires board != null, g != null, color != null
454             * @effects finds the last move and calls paintSquare over the
455             * destination
456             */
457            private void paintLastMove(Graphics g, Color color, ChessBoard board){
458                    if (board.getMoveHistory()!= null){
459                            if (board.getMoveHistory().getLastMove() != null){
460                                    paintSquare(board.getMoveHistory().getLastMove().getPiece().getRow(),board.getMoveHistory().getLastMove().getPiece().getColumn(),
461                                                    color, g, board);
462                                    paintSquare(board.getMoveHistory().getLastMove().getRow(),board.getMoveHistory().getLastMove().getColumn(),
463                                                    color, g, board);
464                            }else{
465                                    List<ChessMove> movesList = board.getMoveHistory().getHistory();
466                                    int size = movesList.size();
467                                    if (size > 0){
468                                            for (int i = size-1; i >=0; i--){
469                                                    if (movesList.get(i)!=null){
470                                                            paintSquare(movesList.get(i).getPiece().getRow(),movesList.get(i).getPiece().getColumn(),
471                                                                            color, g, board);
472                                                            paintSquare(movesList.get(i).getRow(),movesList.get(i).getColumn(),
473                                                                            color, g, board);
474                                                    break;
475                                                    }
476                                            }
477                                    }
478                                    
479                            }
480                    }
481            }

484            /**
485             * Draws the viewedBoard onto g. 
486             * @requires != null
487             */
488            public void paint(Graphics g) {
489                    //sets the board to be painted as the viewed board
490                    ChessBoard paintedBoard = viewedBoard;
491                    Dimension size = getSize();
492                    Image boardImage = imageMap.getImage("board");
493            g.drawImage(boardImage, 
494                        0, 0, size.width, size.height,
495                        0, 0, boardImage.getWidth(null),boardImage.getHeight(null),
496                        null);  
497            
498            //draws what is necessary for disco mode
499            if (dance){
500                    for (int i=0; i<8;i++){
501                            for (int j=0; j<8;j++){
502                                    this.paintSquare(i, j, new Color(255*rand.nextInt(2),255*rand.nextInt(2),255*rand.nextInt(2)), g, paintedBoard);
503                            }
504                    }
505            }
506            
507            //if there is a piece selected, draw the square the piece is on blue, and
508            //possible moves for the piece green
509            if (hasFirstSquare){
510                    if (!((startRow[0]== NULLCOOR)||(startColumn[0]==NULLCOOR))){
511                            
512                            paintSquare(startRow[0],startColumn[0],new Color(0, 0, 255, 150),g, paintedBoard);
513                            if (showPossibleMoves){
514                                    paintPossibleMoves(startRow[0],startColumn[0],new Color(0, 255, 0, 150),g, paintedBoard);
515                            }
516                    }
517                    
518            }
519            
520            if (tabPressed){
521                    paintHasMoves(new Color(0,255,50,220), g, paintedBoard);
522            }
523            
524            if (showLastMove){
525                    paintLastMove(g, new Color(255,102,51,230), paintedBoard);
526            }
527            
528            //Draw the pieces
529            paintPieces(g, paintedBoard);
530            
531            //If the viewedBoard says the game is over draw this
532            if (paintedBoard.isGameOver()){
533                    Font font = new Font("Serif", Font.BOLD, 36);
534                g.setFont(font); 
535                Color color = new Color(255, 0, 255);
536                g.setColor(color);
537                if (paintedBoard.getWinner().equals(Player.NONE)){
538                    if (paintedBoard.getGameOverReason() == Board.FORCEDEND){
539                            g.drawString("Game Over: Ended", boardXOffset, boardYOffset-3);
540                    }else{
541                            g.drawString("Game Over: Stalemate", boardXOffset, boardYOffset-3);
542                    }
543                }else{
544                    g.drawString("Game Over: "+paintedBoard.getWinner().toString().substring(0, 1)+
545                                    paintedBoard.getWinner().toString().toLowerCase().substring(1)+" Wins", boardXOffset, boardYOffset-3);
546                }
547                
548             }
549            
550            if(kitty){ 
551            try{
552                    Image kittyImage = imageMap.getImage("kitty");
553                    g.drawImage(kittyImage, 
554                        0, 0, size.width, size.height,
555                        0, 0, kittyImage.getWidth(null),kittyImage.getHeight(null),
556                        null); 
557                            }catch (Exception ex){}
558            }
559            
560            }

621             * @modifies startLocation, endLocation
622             * @effects
623             *              If acceptingInput == true{
624             *                      if the location clicked on contains a piece of 
625             *                      else
626             *                              sets endLocation to the square that was clicked on
627             *                              calls postMove with the move generated from startLocation,
628             *                              and endLocation 
629             *                              resets startLocation, resets endLocation
630             * 
631             * 
632             */
633            public void mousePressed(MouseEvent e) {
634                            requestFocusInWindow();
635                            //System.out.println("X: "+e.getX());
636                            //System.out.println("Column: "+xToColumn(e.getX()));
637                            //System.out.println("Y: "+e.getY());
638                            //System.out.println("Row: "+ yToRow(e.getY()));
639                            
640                            int row =  yToRow(e.getY());
641                            int column = xToColumn(e.getX());
642                            if (!(( row == NULLCOOR) || (column ==NULLCOOR))){
643                                    if (acceptingInput){
644                                            
645                                            //determines if there is not already a piece
646                                            //if the space clicked on has a piece 
647                                            //and if that piece belongs to the current player
648                                            if( (viewedBoard.getPieceAt(row, column)!=null)
649                                                    &&(viewedBoard.getPieceAt(row, column).getPlayer().equals(currentPlayer.getPlayerType())))
650                                                    {
651                                                    //System.out.println("Has First Square");
652                                                    startRow[0] = yToRow(e.getY());
653                                                    startColumn[0] = xToColumn(e.getX());
654                                                    hasFirstSquare = true;
655                                                    this.postRefresh();
656                                            }
657                                            //checks to see if we have a first square that is a legitimate value
658                                            else if (startRow[0]!= NULLCOOR &&
659                                                              startColumn[0] != NULLCOOR && hasFirstSquare){
660                                                    endRow[0]=yToRow(e.getY());
661                                                    endColumn[0] = xToColumn(e.getX());
662                                                    if( currentPlayer.setMove(startRow, startColumn, 
663                                                                    endRow, endColumn)){
664                                                            this.setNotAcceptingInput();
665                                                            currentPlayer.notifyHasMove();
666                                                    }
667                                                    hasFirstSquare=false;
668                                                    this.postRefresh();
669                                            }
670                                    }
671                            }
672                            
673                            
674            
675            }

677            /**
678            * @returns the column a pixel with an x value would be in
679            *           NULLCOOR if none
680            */           
681            private int xToColumn(int x){
682                    if ((x<boardXOffset)||(x>boardXOffset+boardWidth)){
683                            return NULLCOOR;
684                    }
685                    
686                    return (int) Math.floor((x-1-boardXOffset)/squareWidth);
687                    
688            }
689            /**
690            * @returns the row a pixel with an y value would be in
691            *           NULLCOOR if none
692            */     
693            private int yToRow(int y){
694                    if ((y<boardYOffset)||(y>boardYOffset+boardHeight)){
695                            return NULLCOOR;
696                    }
697                    
698                    return  viewedBoard.getRows()-1 - (int)Math.floor((y-1-boardYOffset)/squareHeight);
699                    
700            }

703            /**
704             * @requires pieceImage != null, 0 <= column < viewedBoard.getColumns()
705             * @return the x pixel that the piece needs to be rendered in
706             */
707            private int getPiecePixelX(int column, Image pieceImage){
708                    //gets the left most pixel for the column
709                    int leftPixel = squareWidth*column+boardXOffset;
710                    //adds the appropriate offset
711                    return leftPixel+squareWidth/2-pieceImage.getWidth(null)/2;
712            }
713            
714            /**
715             * @requires pieceImage != null, 0 <= column < viewedBoard.getRows()
716             * @return the y pixel that the piece needs to be rendered in
717             */
718            private int getRowPixel(int row, Image pieceImage){
719                    //gets the bottom pixel of the square that contains this piece
720                    int topPixel = (viewedBoard.getRows()-row)*squareWidth+boardYOffset;
721                    //adds the appropriate offset
722                    return topPixel-pieceImage.getHeight(null)-squareHeight/10;
723            }
