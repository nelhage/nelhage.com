194                    this.clock = clock;
195                    
196                    currentBoard.removeObserver(this);
197                    this.view.switchViewedBoard(newBoard);
198                    currentBoard = newBoard;

203                    mainFrame.setVisible(true);
204                    
205                    checkRep();
206            }

242            /**
243             * Private method used by the ChessGUI that checks for end game conditions
244             * and displays the proper dialog when end conditions have been met.
245             *
246             */
247            
248            private void checkForEndGame(){
249                    
250                    if (currentBoard.isGameOver() && !notifiedGameOver){
251                            String message;
252                            if (currentBoard.getWinner().equals(Player.WHITE)){
253                                    message = "Game Over: White Wins";
254                            }else if (currentBoard.getWinner().equals(Player.BLACK)){
255                                    message = "Game Over: Black Wins";
256                            }else{
257                                    message = "Game Over: Stalemate";
258                            }
259                            notifiedGameOver = true;
260                            JOptionPane.showMessageDialog(mainFrame, message);
261                    }
262                    
263                    
264            }

468            /**
469             * Handles these events for the ChessGUI:
470             * Exiting
471             * Saving       
472             * Loading
473             * Starting a new game
474             * Pausing
475             * Ending the game
476             * Setting Possible Moves Visible
477             * Disco Mode
478             * 
479             * For a better description of these see the specifications for the class.
480             * 
481             * 
482             * 
483             */
484    
485            public void actionPerformed(ActionEvent e) {
486                    if (e.getActionCommand().equals("Exit")){
487                            System.exit(0);
488                    }else if (e.getActionCommand().equals("NewGame")){
489                            master.pauseGame();
490                            NewGameWindow.createNewGameWindow(master);
491                            pauseButton.setSelected(false);
492                            isPaused = false;
493                            master.unpauseGame();
494                    }else if (e.getActionCommand().equals("SaveGame")){
495                            
496                            
497                            master.pauseGame();
498                            int returnVal = fileChooser.showSaveDialog(mainFrame);
499                            if (returnVal == JFileChooser.APPROVE_OPTION) {
500                                    File file = fileChooser.getSelectedFile();
501                                    try{
502                                            master.save(file);
503                                    }catch(Exception ex){
504                                            JOptionPane.showMessageDialog(mainFrame, "Failed to Save Game");
505                                    }
506                                    
507                            }
508                            pauseButton.setSelected(false);
509                            isPaused = false;
510                            master.unpauseGame();
511                    }else if (e.getActionCommand().equals("LoadGame")){
512                            
513                            master.pauseGame();
514                            int returnVal = fileChooser.showOpenDialog(mainFrame);
515    
516                            if (returnVal == JFileChooser.APPROVE_OPTION) {
517                                    try{
518                                            File file = fileChooser.getSelectedFile();
519                                            LoadGameWindow.createLoadGameWindow(file, master, mainFrame);
520                                    }catch(Exception ex){
521                                            javax.swing.SwingUtilities.invokeLater(new Runnable() {
522                                                    public void run() {
523                                                            JOptionPane.showMessageDialog(mainFrame, "Failed to Load Game");
524                                                }
525                                            });
526                                            
527                                    }
528                            }
529                            pauseButton.setSelected(false);
530                            isPaused = false;
531                            master.unpauseGame();
532                    }else if (e.getActionCommand().equals("PauseGame")){
533                            if (isPaused){
534                                    master.unpauseGame();
535                                    isPaused = false;
536                            }else{
537                                    master.pauseGame();
538                                    isPaused = true;
539                                    view.setNotAcceptingInput();
540                            }
541                    }else if (e.getActionCommand().equals("disco")){
542                            if (disco){
543                                    view.stopDisco();
544                            }else{
545                                    view.startDisco();
546                            }
547                            disco = !disco;
548                    }else if (e.getActionCommand().equals("moves")){
549                            if (possibleMoves){
550                                    view.noShowMoves();
551                            }else{
552                                    view.showMoves();
553                            }
554                            possibleMoves = !possibleMoves;
555                    }else if (e.getActionCommand().equals("resign")){ 
556                            if (!currentBoard.isGameOver())
557                            {
558                                    this.notifiedGameOver = true;
559                                    this.currentBoard.endGame(Board.FORCEDEND, Player.NONE);
560                            }
561                    }
562             
563             
564            }

568              * Creates a ChessGUI suitable for Chess or Antichess playing on.  
569              * 
570              * For the GUI returned:
571              *     board is the board that this GUI is monitoring
572              *     view is the boardView that this GUI uses
573              *     clock is the clock that this GUI monitors
574              *     master is the ControllerMaster directing this GUI
575              *             isPaused = false;
576              *     notifiedGameOver = false;
577              * @return a ChessGUI
578              */

580             public static ChessGUI createGUI(ChessBoard board, GameClock gameClock, ControllerMaster master){
581    //              initializes the BoardView
582                            ArrayList<String> chessNameList = new ArrayList<String>();
583                            chessNameList.add("RAntiking");
584                            chessNameList.add("RBishop");
585                            chessNameList.add("RKing");
586                            chessNameList.add("RKnight");
587                            chessNameList.add("RPawn");
588                            chessNameList.add("RQueen");
589                            chessNameList.add("RRook");
590                            chessNameList.add("WAntiking");
591                            chessNameList.add("WBishop");
592                            chessNameList.add("WKing");
593                            chessNameList.add("WKnight");
594                            chessNameList.add("WPawn");
595                            chessNameList.add("WQueen");
596                            chessNameList.add("WRook");
597                            chessNameList.add("board");
598                            
599                            ImageMap chessMap = new ImageMap("images");
600                            chessMap.loadImageList(chessNameList);
601                            chessMap.loadImage("kitty", "kitty.jpg");
602                            ChessBoardView boardView = new ChessBoardView(board, chessMap, 50, 50, 400, 400);
603                            
604                            //Create and set up the content pane and menu bar
605                            ChessGUI gui = new ChessGUI(boardView, gameClock, master, board);
606                    return gui;
607             }

626
