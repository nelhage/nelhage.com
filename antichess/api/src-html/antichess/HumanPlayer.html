001    package antichess;
002    
003    
004    /**
005     * This interface handles interactions between the Controller and the View
006     * The HumanPlayer facilitates the communication between the GameController and the
007     * BoardView class.  The typical communication bewtween the classes works in this manner:
008     * 
009     * controller calls getMove.  The HumanPlayer tells the BoardView to acceptInput.  When the BoardView
010     * has input it notifies the HumanPlayer.  The HumanPlayer checks for legality in the move.  If the 
011     * move is legal it notifies the controller that it has a move.
012     * 
013     * 
014     * @author nlharr
015     * 
016     * @specfield board             //Board stores the board the HumanPlayer is interacting with
017     * @specfield controller        //controller specifies the GameController that this HumanPlayer is controlled by
018     * @specfield boardView         //BoardView that the HumanPlayer asks for moves
019     * @specfield currentMove       //stores the currentMove to be updated to by the GameController
020     */
021    public interface HumanPlayer<M extends Move> extends GamePlayer<M>{
022            
023            
024            /**
025             * 
026             * @return a valid move if the user has inputed one
027             */
028            public M getMove(GameController controller);
029                    
030            
031            
032            /**
033             * @returns playerType
034             */
035            public Player getPlayerType();
036            
037            /**
038             * @requires startRows, startColumns, destinationRows, destinationColumnes are all not null 
039             * @modifies currentMove
040             * @effects if the move between the start and destination is legal
041             *                      it sets currentMove to that move
042             *
043             * @return true if the move between the start and destination is legal
044             *                      else false
045             */
046            public boolean setMove(int[] startRows, int[] startColumns,
047                                                                                    int[] destinationRows, int[] destinationColumns);
048            
049            /**
050             * Notifies the controller that requested a move if it has a move
051             * @throws RuntimeException if no move has been requested so there is
052             *                      no controller to return a value to
053             */
054            public void notifyHasMove();
055            
056     
057            
058    }
