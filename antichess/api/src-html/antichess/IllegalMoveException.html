003    package antichess;
004    
005    /**
006     * An <tt>Exception</tt> thrown when a client asks a <tt>Board</tt> to
007     * make an illegal move.
008     *
009     * @specfield move : Move // The illegal move 
010     **/
011    public class IllegalMoveException extends Exception
012    {
013        private Move move;     // Ve do not believe in you
014    
015        /**
016         * @effects create an IllegalMoveException with the given Move
017         */
018        public IllegalMoveException(Move m) {
019            super("Illegal move: " + m.toString());
020            this.move = m;
021        }
022    
023        /**
024         * @return this.move
025         */
026        public Move getMove() {
027            return move;
028        }
029    
030            private static final long serialVersionUID = 3649487086528036056L;
031    }
