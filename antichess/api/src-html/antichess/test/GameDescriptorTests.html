032                    super();
033                    descriptors = new GameDescriptor[8];
034                    players0 = new ArrayList<Player>();
035                    players0.add(Player.WHITE);
036                    players0.add(Player.BLACK);
037                    players1 = new ArrayList<Player>();
038                    players1.add(Player.BLACK);
039                    players1.add(Player.WHITE);
040                    
041                    playerDescriptions0 = new ArrayList<String>();
042                    playerDescriptions0.add("Human");
043                    playerDescriptions0.add("AI");
044                    
045                    playerDescriptions1 = new ArrayList<String>();
046                    playerDescriptions1.add("AI");
047                    playerDescriptions1.add("Human");
048                    
049                    playerDescriptions2 = new ArrayList<String>();
050                    playerDescriptions2.add("AI");
051                    playerDescriptions2.add("AI");
052                    
053                    playerDescriptions3 = new ArrayList<String>();
054                    playerDescriptions3.add("Human");
055                    playerDescriptions3.add("Human");
056                    
057                    times0 = new long[2];
058                    times1 = new long[2];
059                    times2 = new long[2];

063                    times0[1] = 0;

078                                    playerDescriptions0, times0, true);

080                    descriptors[1] = new GameDescriptor("Antichess", players1,

082                    
083                    descriptors[2] = new GameDescriptor("Chess", players0, 
084                                    playerDescriptions2, times2, true);
085            
086                    descriptors[3] = new GameDescriptor("Antichess", players1, 
087                                    playerDescriptions3, times3, true);
088            
089                    descriptors[4] = new GameDescriptor("Antichess", players0, 
090                                    playerDescriptions0, times0, false);
091            
092                    descriptors[5] = new GameDescriptor("Chess", players1, 
093                                    playerDescriptions1, times1, false);
094                    
095                    descriptors[6] = new GameDescriptor("Antichess", players0, 
096                                    playerDescriptions2, times2, false);

168            /**
169             * Tests that illegal argument exceptions are thrown for invalid input
170             */
171            public void testExceptionsThrown(){
172                    boolean missedException;
173                    
174                    missedException = true;
175                    try{
176                            new GameDescriptor("Cow", players0, 
177                                            playerDescriptions0, times0, true);
178                    }catch (IllegalArgumentException ex){
179                            missedException = false;
180                    }if (missedException){
181                            fail("Exception not thrown for improper GameTypeName");
182                    }
183                    
184                    List<Player>badPlayers = new ArrayList<Player>();
185                    badPlayers.add(Player.NONE);
186                    badPlayers.add(Player.RED);
187                    
188                    missedException = true;
189                    try{
190                            new GameDescriptor("Chess", badPlayers, 
191                                            playerDescriptions0, times0, true);
192                    }catch (IllegalArgumentException ex){
193                            missedException = false;
194                    }if (missedException){
195                            fail("Exception not thrown for improper Players");
196                    }
197                    
198                    List<String>badPlayerDescriptions = new ArrayList<String>();
199                    badPlayerDescriptions.add("Human");
200                    badPlayerDescriptions.add("Monkey");
201                    
202                    missedException = true;
203                    try{
204                            new GameDescriptor("Chess", players0, 
205                                            badPlayerDescriptions, times0, true);
206                    }catch (IllegalArgumentException ex){
207                            missedException = false;
208                    }if (missedException){
209                            fail("Exception not thrown for improper playerDescriptions");
210                    }
211                    
212                    long[] badTimes = {-52, 500};
213                    
214                    missedException = true;
215                    try{
216                            new GameDescriptor("Chess", players0, 
217                                            playerDescriptions0, badTimes, true);
218                    }catch (IllegalArgumentException ex){
219                            missedException = false;
220                    }if (missedException){
221                            fail("Exception not thrown for improper playerTimes");
222                    }
223                    
224                    playerDescriptions0.add("Human");
225                    
226                    missedException = true;
227                    try{
228                            new GameDescriptor("Chess", players0, 
229                                            playerDescriptions0, times0, true);
230                    }catch (IllegalArgumentException ex){
231                            missedException = false;
232                    }if (missedException){
233                            fail("Exception not thrown for lists of differing sizes");
234                    }
235                    
236                    
237                    
238            }
