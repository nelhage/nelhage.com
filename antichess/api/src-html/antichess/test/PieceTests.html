003    package antichess.test;
004    
005    import java.util.*;
006    import junit.framework.*;
007    import antichess.*;
008    
009    /**
010     * Tests for the Piece class
011     **/
012    public class PieceTests extends TestCase
013    {
014            PieceType pieceType = new NullPieceType();
015            
016            public PieceTests(String name) {
017                    super(name);
018            }
019    
020            private void tryPiece(Player p, PieceType pt, int r, int c) {
021                    Piece piece = new Piece(p, pt, r, c);
022                    tryPiece(piece, p, pt, r, c);
023            }
024    
025            private void tryPiece(Piece piece, Player p, PieceType pt, int r, int c) {
026                    assertEquals(piece.getPlayer(), p);
027                    assertEquals(piece.getType(), pt);
028                    assertEquals(piece.getRow(), r);
029                    assertEquals(piece.getColumn(), c);
030            }
031    
032            public void testConstructor() {
033                    tryPiece(Player.WHITE, pieceType, 0, 0);
034                    tryPiece(Player.WHITE, pieceType, -1, -1);
035                    tryPiece(Player.BLACK, pieceType, -1, -1);
036                    tryPiece(Player.BLACK, pieceType, 2, 4);
037            }
038    
039            public void testMove() {
040                    Piece piece = new Piece(Player.WHITE, pieceType, 0, 0);
041                    tryPiece(piece.moveTo(4, 4), Player.WHITE, pieceType, 4, 4);
042    
043                    piece = new Piece(Player.BLACK, pieceType, 5, 5);
044                    tryPiece(piece.moveTo(-1, -1), Player.BLACK, pieceType, -1, -1);
045            }
046    
047            public void testImmutable() {
048                    Piece piece = new Piece(Player.WHITE, pieceType, 0, 0);
049                    piece.moveTo(4, 4);
050                    tryPiece(piece, Player.WHITE, pieceType, 0, 0);
051    
052                    piece = new Piece(Player.BLACK, pieceType, 5, 5);
053                    piece.moveTo(-1, -1);
054                    tryPiece(piece, Player.BLACK, pieceType, 5, 5);
055            }
056    
057            public void testEquals() {
058                    Piece p = new Piece(Player.WHITE, pieceType, 0, 0);
059                    assertEquals(p, new Piece(Player.WHITE, pieceType, 0, 0));
060                    assertEquals(p, p);
061                    assertFalse(p.equals(null));
062                    assertFalse(p.equals(new Piece(Player.BLACK, pieceType, 0, 0)));
063                    assertFalse(p.equals(new Piece(Player.WHITE, pieceType, 1, 0)));
064                    assertFalse(p.equals(new Piece(Player.WHITE, pieceType, 0, 1)));
065                    assertFalse(p.equals(new Piece(Player.WHITE, new NullPieceType(), 0, 1)));
066            }
067    }
