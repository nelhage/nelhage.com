356            /**
357             * Private method to verify that the chessboard has pieces in the expected
358             * locations, both on the board and in the captured piles. This tests
359             * the getPieceAt() method and getCapturePieces() methods.
360             * @param cb the chessboard
361             * @param pieces the list of pieces, with their expected locations on the board
362             * and in the captured pile.
363             */
364            private void checkPieceLocations(ChessBoard cb, List<Piece> pieces) {
365                    Piece test;
366                    boolean found;
367                    // Test that each of the pieces in the given list is in the correct position
368                    for (Piece p : pieces) {
369                            if (p.getRow() != -1 && p.getColumn() != -1) {
370                                    test = cb.getPieceAt(p.getRow(), p.getColumn());
371                                    assertEquals("Incorrect piece found at " + p.getRow() + ", " + p.getColumn(),
372                                                    p, test);
373                            } else if (p.getRow() == -1 && p.getColumn() == -1) {
374                                    found = false;
375                                    List<Piece> capturedPile = cb.getCapturedPieces();
376                                    for (Piece cp : capturedPile)
377                                            if (p.equals(cp))
378                                                    found = true;
379                                    assertTrue("Expected captured piece not found in captured pile", found);
380                            } else {
381                                    fail("Expected piece had an invalid location");
382                                    return;
383                            }
384                    }
385            }

387            /**
388             * Checks that the pieces for a given player (or both players if player == null)
389             * have the expected locations. This allows for verification of the getPieces()
390             * and getPieces(player) methods.
391             * @param cb the chessboard
392             * @param pieces the list of expected pieces for the given player
393             * @param player the player in question.
394             */
395            private void checkGetPieces(ChessBoard cb, List<Piece> pieces, Player player) {
396                    List<Piece> piecesCopy = new ArrayList<Piece>(pieces);
397                    List<Piece> currentPieceList;
398                    
399                    if (player == null)
400                            currentPieceList = cb.getPieces();
401                    else
402                            currentPieceList = cb.getPieces(player);
403                    
404                    for (Piece nextPiece : currentPieceList) {
405                            assertTrue("Board contains an unexpected piece: " + nextPiece.toString(),
406                                            piecesCopy.contains(nextPiece));
407                            piecesCopy.remove(nextPiece);
408                    }
409                    
410                    if (player == null)
411                            currentPieceList = cb.getCapturedPieces();
412                    else
413                            currentPieceList = cb.getCapturedPieces(player);
414                    
415                    for (Piece nextPiece : currentPieceList) {
416                            assertTrue("Board's captured pile contains an unexpected piece: " + nextPiece.toString(),
417                                            piecesCopy.contains(nextPiece));
418                            piecesCopy.remove(nextPiece);
419                    }
420                    
421                    assertTrue("Board did not contain all of the given pieces on the board or in the captured pile",
422                                    piecesCopy.size() == 0);
423            }

427             * @param cb the chessboard
428             * @param expected the expected movehistory
429             */
430            private void checkGetMoveHistory(ChessBoard cb, MoveHistory<ChessMove> expected) {
431                    if (expected == null)

433                                            cb.getMoveHistory() == null);
434                    else {
435                            List<ChessMove> expectedHistory = expected.getHistory();
436                            List<String> expectedTimes = expected.getMoveTimes();
437                            List<ChessMove> actualHistory = cb.getMoveHistory().getHistory();
438                            List<String> actualTimes = cb.getMoveHistory().getMoveTimes();
439                            
440                            assertEquals("Expected move history does not match actual move history.",
441                                            expectedHistory, actualHistory);
442                            assertEquals("Expected move times do not match actual move times.",
443                                            expectedTimes, actualTimes);
444                    }
445            }

487            /**
488             * Helper method to execute a ChessMove on this board and update
489             * the list of pieces correctly. ChessMove <code>cm</code> must
490             * be a legal move for this chessboard.
491             * @param cb the chessboard
492             * @param cm the ChessMove to execute
493             * @param start the original list of pieces on the board to update
494             * @param timestamp the timestamp to associate with <code>cm</code>
495             * @return a new list of pieces with the piece locations properly
496             * updated.
497             */
498            private List<Piece> doMove(ChessBoard cb, ChessMove cm, List<Piece> start, String timestamp) {
499                    List<Piece> end = new ArrayList<Piece>(start);
500                    int pieceToMoveIndex = start.indexOf(cm.getPiece());
501                    int capturedPieceIndex;
502                    Piece p;
503                    
504                    try {
505                            if (timestamp.equals(""))
506                                    cb.doMove(cm);
507                            else
508                                    cb.doMove(cm, timestamp);
509                    } catch (Exception e) {
510                            fail("Board disallowed a legal move: " + e.getMessage());
511                            return end;
512                    }
513                    
514                    end.set(pieceToMoveIndex, end.get(pieceToMoveIndex).moveTo(cm.getRow(), cm.getColumn()));
515                    p = end.get(pieceToMoveIndex);
516                    // Take care of pawn promotions
517                    if (p.getType().equals(PieceTypeFactory.getPawnType()) && 
518                                    ((p.getPlayer().equals(Player.WHITE) && p.getRow() == 7) ||
519                                     (p.getPlayer().equals(Player.BLACK) && p.getRow() == 0)))
520                            end.set(pieceToMoveIndex, new Piece(p.getPlayer(), PieceTypeFactory.getQueenType(), p.getRow(), p.getColumn()));
521                    
522                    if (cm.getCapturedPiece() != null) {
523                            capturedPieceIndex = start.indexOf(cm.getCapturedPiece());
524                            end.set(capturedPieceIndex, end.get(capturedPieceIndex).moveTo(-1, -1));
525                    }
526                    return end;
527            }

529            /**
530             * Helper method to undo the last ChessMove executed on a given
531             * chessboard and update the list of pieces properly. At least
532             * one move must have been executed on the given chessboard and
533             * recorded in the board's move history.
534             * @param cb the chessboard
535             * @param start the list of pieces with their current locations
536             * on the board.
537             * @return a new list of pieces with the piece locations properly
538             * updated.
539             */
540            private List<Piece> undoMove(ChessBoard cb, List<Piece> start) {
541                    List<Piece> end = new ArrayList<Piece>(start);
542                    // Grab the last move from the history
543                    List<ChessMove> history = cb.getMoveHistory().getHistory();
544                    ChessMove last = history.get(history.size()-1);
545                    
546                    int pieceMovedIndex = start.indexOf(last.getPiece().moveTo(last.getRow(), last.getColumn()));
547                    if (pieceMovedIndex == -1) // Pawn Promotion
548                            pieceMovedIndex = start.indexOf(new Piece(last.getPiece().getPlayer(), PieceTypeFactory.getQueenType(), last.getRow(), last.getColumn()));
549                    int capturedPieceIndex;
550                    
551                    try {
552                            cb.undoLastMove();
553                    } catch (Exception e) {
554                            fail("Board disallowed undoing move: " + e.getMessage());
555                            return end;
556                    }
557                    
558                    end.set(pieceMovedIndex, last.getPiece());
559                    if (last.getCapturedPiece() != null) {
560                            capturedPieceIndex = start.indexOf(last.getCapturedPiece().moveTo(-1, -1));
561                            end.set(capturedPieceIndex, last.getCapturedPiece());
562                    }
563                    return end;
564            }

613             /**
614              * Helper method to recursively check the self-consistency of
615              * the board, verifying that moves returned by allLegalMoves()
616              * can be made with the expected consequences.
617              * @param cb the chessboard
618              * @param pieces the list holding the arrangement of pieces
619              * on the given chessboard
620              * @param i the level of recursion at which this method is
621              * currently running. Will be less than MAX_RECURSE.
622              */
623            private void checkSelfConsistency(ChessBoard cb, List<Piece> pieces, int i) {
624                    ChessBoard localcb = (ChessBoard)cb.clone();
625                    List<ChessMove> allLegal = localcb.allLegalMoves();
626                    
627                    for (ChessMove cm : allLegal) {
628                            checkMoveLegality(localcb, cm, true);
629                            pieces = doMove(localcb, cm, pieces);
630                            checkPieces(localcb, pieces);
631                            
632                            if (i < MAX_RECURSE)
633                                    checkSelfConsistency(localcb, new ArrayList<Piece>(pieces), i+1);
634                            
635                            pieces = undoMove(localcb, pieces);
636                            checkPieces(localcb, pieces);
637                    }
638            }
