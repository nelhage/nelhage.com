001    package antichess.test;
002    
003    import java.io.InputStreamReader;
004    
005    import junit.framework.TestCase;
006    import antichess.AntichessBoard;
007    import antichess.GameClock;
008    import antichess.Player;
009    import antichess.TextPlayer;
010    
011    public class TextPlayerTests extends TestCase{
012            
013            //creation of one player for testing a few things
014            AntichessBoard board1;
015            GameClock clock1;
016            TextPlayer p1;
017            TextPlayer p2;
018            
019            public TextPlayerTests(){
020                    super();
021                    board1 = new AntichessBoard();
022                    board1.newGame();
023                    
024                    
025                    Player[] playerList1 = {Player.WHITE, Player.BLACK};
026                    long[] playerTimes1 = {30000, 30000};
027                    clock1 = new GameClock(playerList1, playerTimes1);
028                    
029                    
030            }
031            
032            
033            private void initializePlayers(){
034                    p1 = new TextPlayer(board1, clock1, Player.WHITE);
035                    p2 = new TextPlayer(board1, clock1, Player.BLACK);
036            }
037            
038            /**
039             * Tests that constructor doesn't throw an exception with valid
040             * values
041             */
042            public void testConstructor(){
043                    try{
044                            initializePlayers();
045                    }catch (Exception ex){
046                            fail("Threw exceptiong "+ex.getMessage()+" while creating legal TextPlayers");
047                    }
048            }
049            
050            /**
051             * Tests that automaticallyTimed is false
052             */
053            public void testAutomaticallyTimed(){
054                    initializePlayers();
055                    assertFalse(p1.automaticallyTimed());
056                    assertFalse(p2.automaticallyTimed());
057            }
058            
059            /**
060             * Tests getPlayerType for BLACK and WHITE TextPlayers
061             */
062            public void testGetPlayerType(){
063                    initializePlayers();
064                    assertTrue(p1.getPlayerType().equals(Player.WHITE));
065                    assertTrue(p2.getPlayerType().equals(Player.BLACK));
066            }
067            
068            
069            /**
070             * Tests the stringToMove method on a variety of moves available with
071             * a starting board.
072             */
073            public void testStringToMove(){
074                    initializePlayers();
075                    assertEquals(board1.createMove(board1.getPieceAt(1, 1), 3, 1),   TextPlayer.stringToMove("b2-b4", board1));
076                    assertEquals(board1.createMove(board1.getPieceAt(0, 1), 2, 2),   TextPlayer.stringToMove("b1-c3", board1));
077                    assertEquals(board1.createMove(board1.getPieceAt(1, 7), 2, 7),   TextPlayer.stringToMove("h2-h3", board1));
078                    assertEquals(board1.createMove(board1.getPieceAt(0, 6), 2, 5),   TextPlayer.stringToMove("g1-f3", board1));
079                    try{
080                            board1.doMove(board1.createMove(board1.getPieceAt(1, 1), 3, 1));
081                    }catch (Exception ex){
082                            fail("Board did not perform a legal move");
083                    }
084                    assertEquals(board1.createMove(board1.getPieceAt(6, 1), 4, 1),   TextPlayer.stringToMove("b7-b5", board1));
085                    assertEquals(board1.createMove(board1.getPieceAt(7, 6), 5, 7),   TextPlayer.stringToMove("g8-h6", board1));
086                    try{
087                            board1.doMove(board1.createMove(board1.getPieceAt(6, 1), 4, 1));
088                    }catch (Exception ex){
089                            fail("Board did not perform a legal move");
090                    }
091                    
092            }
093            
094            /**
095             * doMove() is tested through the TextUI tests
096             * getPossibleMove is tested through the TextUI tests
097             * doPossibleMove is tested through the TextUI tests
098             */
099            
100            
101                    
102    }
