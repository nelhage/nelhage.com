001    package antichess.test;
002    
003    import junit.framework.TestCase;
004    import antichess.*;
005    
006    /**
007     * Automated tests for ChessMoves.
008     */
009    public class ChessMoveTests extends TestCase
010    {
011            private Piece whitePiece, blackPiece;
012            
013            public ChessMoveTests(String name) {
014                    super(name);
015                    
016                    PieceType pieceType = new NullPieceType();
017                    whitePiece = new Piece(Player.WHITE, pieceType, 0, 0);
018                    blackPiece = new Piece(Player.BLACK, pieceType, 7, 5);
019            }
020            
021            /**
022             * Tests that the Constructor can create ChessMoves with the expected values
023             * when given legal inputs, withough unexpected consequences.
024             */
025            public void testConstructor() {
026                    ChessMove cm;
027                    try {
028                            // Test for a basic ChessMove
029                            cm = new ChessMove(whitePiece, 1, 1);
030                    } catch (Exception e) {
031                            fail("ChessMove constructor threw Exception rather than accepting valid input.");
032                            return;
033                    }
034                    checkChessMove(cm, whitePiece, 1, 1, null);
035                    
036                    try {
037                            // Test a ChessMove with an explicitly "null" captured piece
038                            cm = new ChessMove(blackPiece, 6, 3, null);
039                    } catch (Exception e) {
040                            fail("ChessMove constructor threw Exception rather than accepting valid input.");
041                            return;
042                    }
043                    checkChessMove(cm, blackPiece, 6, 3, null);
044                    
045                    try {
046                            // Test a ChessMove with a non-null captured piece
047                            cm = new ChessMove(whitePiece, 6, 3, blackPiece.moveTo(6, 3));
048                    } catch (Exception e) {
049                            fail("ChessMove constructor threw Exception rather than accepting valid input.");
050                            return;
051                    }
052                    checkChessMove(cm,whitePiece, 6, 3, blackPiece.moveTo(6, 3));
053            }
054            
055            /**
056             * Tests the equality between ChessMoves, verifying that ChessMoves that
057             * should be equal are equal, unequal ChessMoves are unequal, and
058             * ChessMoves aren't equal to non-ChessMoves.
059             */
060            public void testEquals() {
061                    // Standard ChessMove
062                    ChessMove cm = new ChessMove(whitePiece, 1, 1);
063                    assertTrue("ChessMove does not equal itself",
064                                    cm.equals(cm));
065                    assertTrue("ChessMove does not equal equivalent ChessMove",
066                                    cm.equals(new ChessMove(whitePiece, 1, 1)));
067                    assertFalse("Valid ChessMove equivalent to non-valid ChessMove",
068                                    cm == null);
069                    assertFalse("Valid ChessMove equivalent to non-valid ChessMove",
070                                    cm.equals(new Object()));
071                    
072                    // Explicitly null captured piece
073                    ChessMove cm2 = new ChessMove(blackPiece, 6, 3, null);
074                    assertTrue("ChessMove does not equal itself",
075                                    cm2.equals(cm2));
076                    assertTrue("ChessMove does not equal equivalent ChessMove",
077                                    cm2.equals(new ChessMove(blackPiece, 6, 3, null)));
078                    assertFalse("Valid ChessMove equivalent to non-valid ChessMove",
079                                    cm2 == null);
080                    assertFalse("Valid ChessMove equivalent to non-valid ChessMove",
081                                    cm2.equals(new Object()));
082                    assertFalse("ChessMove equivalent to non-equivalent ChessMove",
083                                    cm2.equals(cm));
084                    
085                    // Non-null captured piece
086                    ChessMove cm3 = new ChessMove(whitePiece, 6, 3, blackPiece.moveTo(6, 3));
087                    assertTrue("ChessMove does not equal itself",
088                                    cm3.equals(cm3));
089                    assertTrue("ChessMove does not equal equivalent ChessMove",
090                                    cm3.equals(new ChessMove(whitePiece, 6, 3, blackPiece.moveTo(6, 3))));
091                    assertFalse("Valid ChessMove equivalent to non-valid ChessMove",
092                                    cm3 == null);
093                    assertFalse("Valid ChessMove equivalent to non-valid ChessMove",
094                                    cm3.equals(new Object()));
095                    assertFalse("ChessMove equivalent to non-equivalent ChessMove",
096                                    cm3.equals(cm));
097                    assertFalse("ChessMove equivalent to non-equivalent ChessMove",
098                                    cm3.equals(cm2));
099            }
100            
101            /**
102             * Helper method to verify that a given ChessMove has the expected visible
103             * data.
104             * @param cm the ChessMove
105             * @param p the Piece in the ChessMove
106             * @param row the new row of the Piece of the ChessMove
107             * @param col the new column of the Piece of the ChessMove
108             * @param captured the (potentially null) captured piece in this ChessMove
109             */
110            private void checkChessMove(ChessMove cm, Piece p, int row, int col, Piece captured) {
111                    assertEquals("Unexpected Piece in ChessMove",
112                                    p, cm.getPiece());
113                    assertEquals("Unexpected row in ChessMove",
114                                    row, cm.getRow());
115                    assertEquals("Unexpected column in ChessMove",
116                                    col, cm.getColumn());
117                    assertEquals("Unexpected captured piece in ChessMove",
118                                    captured, cm.getCapturedPiece());
119            }
120    }
