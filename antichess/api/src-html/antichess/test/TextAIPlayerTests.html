001    package antichess.test;
002    
003    import junit.framework.TestCase;
004    import antichess.AntichessBoard;
005    import antichess.GameClock;
006    import antichess.GameController;
007    import antichess.Player;
008    import antichess.TextAIPlayer;
009    
010    public class TextAIPlayerTests extends TestCase{
011            
012            //creation of one player for testing a few things
013            AntichessBoard board1;
014            GameClock clock1;
015            TextAIPlayer p1;
016            TextAIPlayer p2;
017            
018            public TextAIPlayerTests(){
019                    super();
020                    board1 = new AntichessBoard();
021                    board1.newGame();
022                    
023                    
024                    Player[] playerList1 = {Player.WHITE, Player.BLACK};
025                    long[] playerTimes1 = {30000, 30000};
026                    clock1 = new GameClock(playerList1, playerTimes1);
027                    
028                    
029            }
030            
031            
032            private void initializePlayers(){
033                    p1 = new TextAIPlayer(board1, clock1, Player.WHITE, Player.BLACK);
034                    p2 = new TextAIPlayer(board1, clock1, Player.BLACK, Player.WHITE);
035            }
036            
037            /**
038             * Tests that the constructor for a legal value does not through any exceptions
039             */
040            public void testConstructor(){
041                    try{
042                            initializePlayers();
043                    } catch(Exception ex){
044                            fail("Constructor threw error "+ex.getMessage());
045                    }
046            }
047            
048            /**
049             * Tests that automaticallyTimed is false
050             */
051            public void testAutomaticallyTimed(){
052                    initializePlayers();
053                    assertFalse(p1.automaticallyTimed());
054                    assertFalse(p2.automaticallyTimed());
055            }
056            
057            /**
058             * Tests getPlayerType for BLACK and WHITE TextPlayers
059             */
060            public void testGetPlayerType(){
061                    initializePlayers();
062                    assertTrue(p1.getPlayerType().equals(Player.WHITE));
063                    assertTrue(p2.getPlayerType().equals(Player.BLACK));
064            }
065            
066            
067            /**
068             * Tests that doMove returns false
069             */
070            public void testDoMove(){
071                    initializePlayers();
072                    assertFalse(p1.doMove("b2-b4", 10));
073                    assertFalse(p2.doMove("c2-c4", 0));
074            }
075            
076            
077            /**
078             * getPossibleMoves, doPossibleMove and getMove are tested in the TextUI
079             */
080    }
