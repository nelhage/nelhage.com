017            private void initializeMoveTypes() {
018                    pieceMoveTypes = new HashMap<PieceType, MoveGenerator<ChessMove>>();
019                    pieceMoveTypes.put(PieceTypeFactory.getPawnType(),
020                                                       new PawnMoveGenerator());
021                    pieceMoveTypes.put(PieceTypeFactory.getRookType(),
022                                                       new LinearMoveGenerator(false, true));
023                    pieceMoveTypes.put(PieceTypeFactory.getKnightType(),
024                                                       new JumpMoveGenerator(new int[]{1,-1, 1,-1, 2,-2, 2,-2},
025                                                                                                     new int[]{2, 2,-2,-2, 1, 1,-1,-1}));
026                    pieceMoveTypes.put(PieceTypeFactory.getBishopType(),
027                                                       new LinearMoveGenerator(true, false));
028                    pieceMoveTypes.put(PieceTypeFactory.getQueenType(),
029                                                       new LinearMoveGenerator(true, true));
030                    pieceMoveTypes.put(PieceTypeFactory.getKingType(),
031                                                       new UnionMoveGenerator<ChessMove>(
032                                                                                                                             new JumpMoveGenerator(new int[]{ 1, 1, 1, 0, 0,-1,-1,-1},
033                                                                                                                                                                       new int[]{ 1, 0,-1, 1,-1, 1, 0,-1}),
034                                                                                                                             new CastlingMoveGenerator()));
035            }
036    
037            protected MoveGenerator<ChessMove> getMovement(Piece p) {
038                    return pieceMoveTypes.get(p.getType());
039            }
040

043                    initializeMoveTypes();

093            private void testMoves(List<ChessMove> moveList,
094                                                                      Piece piece, Board board,
095                                                                      int[][] moves) {
096                    int count = 0;
097                    boolean ok;
098                    for(ChessMove cm : moveList) {
099                            count++;
100                            assertEquals(cm.getPiece(), piece);
101                            if(!cm.getPiece().getType().equals(PieceTypeFactory.getPawnType())) {
102                                    assertEquals(cm.getCapturedPiece(),
103                                                             board.getPieceAt(cm.getRow(), cm.getColumn()));
104                            }
105                            ok = false;
106                            for(int[] pos : moves) {
107                                    if(pos[0] == cm.getRow() &&
108                                       pos[1] == cm.getColumn())
109                                            ok = true;
110                            }
111                            if(!ok)
112                                    assertTrue( piece + " returned illegal move " + cm, false);
113                    }
114                    assertEquals(piece + " returned wrong number of moves", moves.length, count);
115            }

117            private void testPiece(Board<ChessMove> board,
118                                                       Piece piece,
119                                                       int[][] expectedMoves) {
120                    List<ChessMove> m = getMovement(piece).getMoves(piece, board);
121                    testMoves(m, piece, board, expectedMoves);
122                    for(ChessMove move : m) {
123                            assertTrue("Move " + move + " should be legal",
124                                               getMovement(piece).isMoveLegal(move, board));
125                    }
126            }

138                    board.loadGame(pieces, Player.WHITE, null);
139                    testPiece(board, piece, expectedMoves);
140            }

311            public void testEnPassant() {
312                    ChessBoard board;
313                    Piece white, black, white2;
314                    int [][] expectedMoves;
315    
316                    board = new ChessBoard();
317    
318                    white = white(4,3);
319                    white2 = white(4,5);
320                    black = black(6,4);
321                    board.loadGame(pieces(white,
322                                                              black,
323                                                              white2,
324                                                              whiteKing(0,0),
325                                                              blackKing(7,7)),
326                                               Player.BLACK, null);
327                    
328                    assertEquals(5, board.getPieces().size());
329                    try {
330                            board.doMove(new ChessMove(black, 4, 4));
331                    } catch(IllegalMoveException e) {
332                            fail("black double-move was illegal");
333                    }
334                    assertEquals(5, board.getPieces().size());
335    
336                    expectedMoves = new int[][] {{5,3}, {5,4}};
337                    testPiece(board, white, expectedMoves);
338                    expectedMoves = new int[][] {{5,5}, {5,4}};
339                    testPiece(board, white2, expectedMoves);
340            }

342            /**
343             * Make sure pawns can't capture en passant when they shouldn't be
344             * able able to (e.g. pawns that moved past them with one move, or
345             * other pieces
346             */
347            public void testFalseEnPassant() {
348                    ChessBoard board;
349                    Piece white, black;
350    
351                    board = new ChessBoard();
352    
353                    white = white(4,3);
354                    black = black(5,4);
355                    board.loadGame(pieces(white,
356                                                              black,
357                                                              whiteKing(0,0),
358                                                              blackKing(7,7)),
359                                               Player.BLACK, null);
360                    try {
361                            // Move the black pawn past the white one
362                            board.doMove(board.createMove(black, 4, 4));
363                    } catch(IllegalMoveException ex) {
364                            fail("Move " + ex.getMove() + " was illegal");
365                    }
366                    try {
367                            // Try to take it
368                            board.doMove(board.createMove(white, 5,4));
369                            fail("Pawn was able to en passant a pawn that only moved one square");
370                    } catch(IllegalMoveException ex) {
371                            // OK
372                    }
373    
374                    board.loadGame(pieces(white,
375                                                              new Piece(Player.BLACK,
376                                                                                    PieceTypeFactory.getKnightType(),
377                                                                                    6,3),
378                                                              whiteKing(0,0),
379                                                              blackKing(7,7)),
380                                               Player.BLACK, null);
381                    try {
382                            // Move the black knight next to the white pawn
383                            board.doMove(board.createMove(board.getPieceAt(6,3), 4, 4));
384                    } catch(IllegalMoveException ex) {
385                            fail("Move " + ex.getMove() + " was illegal");
386                    }
387                    try {
388                            // Try to take it
389                            board.doMove(board.createMove(white, 5,4));
390                            fail("Pawn was able to en passant a knight!");
391                    } catch(IllegalMoveException ex) {
392                            // OK
393                    }
394                    
395            }
