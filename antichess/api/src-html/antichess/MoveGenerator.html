003    package antichess;
004    
005    import java.util.List;
006    
007    /**
008     * A MoveGenerator is an abstract representation of the types of moves
009     * allowed by a piece. A MoveGenerator encapsulates local knowledge
010     * about the patterns of movement allowed by a piece (straight-line,
011     * finite distance or unlimited, whether the piece can move through
012     * other pieces, and so on), but not higher-order knowledge such as
013     * ``Would this move cause the player to be in check?'' The Board
014     * class is responsible for maintaining that information.
015     **/
016    public interface MoveGenerator<M extends Move>
017    {
018        /**
019         * @requires p.type == this
020         * @return an interator of moves the given piece can perform on
021         * the board
022         **/
023        public List<M> getMoves(Piece p, Board<M> b);
024    
025        /**
026         * @requires m.piece.type == this
027         * @return true iff the given move is allowed by its piece's
028         * movement types
029         **/
030        public boolean isMoveLegal(M m, Board<M> b);
031    }
