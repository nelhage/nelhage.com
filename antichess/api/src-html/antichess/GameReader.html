035            /**
036             * @effects constructs a new GameReader containing the data in the
037             * given XML stream
038             * @throws InvalidGameFileException if the stream is not a valid game (e.g. it does not validate)
039             */
040            public GameReader(InputStream stream)
041                    throws InvalidGameFileException, IOException {
042                    DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
043                    try {
044                            DocumentBuilder builder = factory.newDocumentBuilder();
045                            Document document = builder.parse(new InputSource(stream));
046                            loadDocument(document);
047                    } catch (SAXException sxe) {
048                            throw new InvalidGameFileException(sxe.toString());
049                    } catch (ParserConfigurationException pce) {
050                            throw new RuntimeException("Couldn't construct a parser: " + pce);
051                    }
052            }

096             * @effects Load the given board with the data parsed from an XML
097             * &lt;pieces&gt; tag, including both the start and end tags, and
098             * set the specified player.
099             * @throws InvalidGameFileException if the XML is invalid
100             */
101            public static void loadBoard(ChessBoard board, String xml, Player player)
102                    throws InvalidGameFileException {
103                    DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
104                    try {
105                            DocumentBuilder builder = factory.newDocumentBuilder();
106                            Document document = builder.parse(new InputSource(new StringReader(xml)));
107                            List<Piece> pieces = parsePieces(document.getDocumentElement());
108                            board.loadGame(pieces, player, null);
109                    } catch (SAXException sxe) {
110                            throw new InvalidGameFileException(sxe.toString());
111                    } catch (ParserConfigurationException pce) {
112                            throw new RuntimeException("Couldn't construct a parser: " + pce);
113                    } catch (IOException ex) {
114                            // This just shouldn't happen. srsly.
115                            throw new RuntimeException("There was an IO error reading from a String.");
116                    }
117            }

119            /**
120             * @effects load a game from the specified DOM Document object
121             */
122            private void loadDocument(Document document)
123                    throws InvalidGameFileException {
124                    Schema schema;
125                    try {
126                            schema = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI).
127                                    newSchema(GameReader.class.getResource("antichess.xsd"));
128                    } catch(SAXException e) {
129                            throw new RuntimeException("Unable to read antichess.xsd: " + e);
130                    }
131                    // This is commented out for the moment since for some reason
132                    // the validator fails to load properly on some computers with
133                    // no useful error message.
134                    try {
135                            Validator validator = schema.newValidator();
136                            validator.validate(new DOMSource(document));
137                    } catch(SAXException e) {
138                            throw new InvalidGameFileException("XML document does not validate: " + e);
139                    } catch(IOException e) {
140                            throw new InvalidGameFileException("IO Error parsing document: " + e);
141                     }
142                    
143                    Node root = getChildOfType(document, Node.ELEMENT_NODE);
144                    if(!root.getNodeName().equals("game"))
145                            throw new InvalidGameFileException("Expected a <game> node");
146                    String ruleset = getAttribute(root, "ruleset");
147                    if(!ruleset.equals(ANTICHESS_RULESET))
148                            throw new InvalidGameFileException("Unknown ruleset: " + ruleset);
149                    XPath xpath = XPathFactory.newInstance().newXPath();
150                    try {
151                            parseTime((Node)xpath.evaluate("/game/time", document, XPathConstants.NODE));
152                            MoveHistory<ChessMove> history = parseMoveHistory((Node)xpath.evaluate("/game/moveHistory", document, XPathConstants.NODE));
153                            List<Piece> pieces = parsePieces((Node)xpath.evaluate("/game/pieces", document, XPathConstants.NODE));
154                            Player player;
155                            player = history == null
156                                    ? Player.WHITE
157                                    : history.getLastMove().getPlayer().otherPlayer();
158                            AntichessBoard bd = new AntichessBoard();
159                            bd.loadGame(pieces, player, history);
160                            // TODO: compare bd with board for sanity
161                            Object gameOver = xpath.evaluate("/game/gameOver", document, XPathConstants.NODE);
162                            if(gameOver != null)
163                                    parseGameOver((Node)gameOver);
164                    } catch(XPathExpressionException e) {
165                            throw new RuntimeException("Internal error: " + e);
166                    } catch(Exception e) {
167                            e.printStackTrace();
168                            throw new InvalidGameFileException(e.toString());
169                    }
170            }

172            /**
173             * @effects Populates whiteTimer and blackTimer based on reading
174             * the given <time> XML node
175             */
176            private void parseTime(Node time) {
177                    isTimed = "true".equals(getAttribute(time, "timed"));
178                    if(!isTimed) return;
179                    long initWhite = Long.parseLong(getAttribute(time, "initWhite"));
180                    long initBlack = Long.parseLong(getAttribute(time, "initBlack"));
181                    long curWhite = Long.parseLong(getAttribute(time, "currentWhite"));
182                    long curBlack = Long.parseLong(getAttribute(time, "currentBlack"));
183                    whiteTimer = new GameTimer(initWhite, curWhite, 500);
184                    blackTimer = new GameTimer(initBlack, curBlack, 500);
185            }

187            /**
188             * @return The MoveHistory represented by the given &lt;moveHistory&gt;
189             * XML element
190             * @effect Populate this.board with the game resulting from
191             * starting with a new game and playing the sequence of moves in
192             * the move history.
193             */
194            private MoveHistory<ChessMove> parseMoveHistory(Node history)
195                    throws InvalidGameFileException {
196                    board = new AntichessBoard();
197                    /* We actually play the game on a board as we load it */
198                    board.newGame();
199                    NodeList children;
200                    MoveHistory<ChessMove> moveHistory = null;
201                    Node node;
202                    ChessMove m;
203    
204                    children = history.getChildNodes();
205                    for(int i=0;i<children.getLength();i++) {
206                            node = children.item(i);
207                            if("move".equals(node.getNodeName())) {
208                                    m = parseMove(node, board);
209                                    if(m.getPlayer() != board.getPlayer()) {
210                                            // Two moves in a row from the same player. This
211                                            // means one player was stalemated and had no
212                                            // legal moves.
213                                            try{
214                                                    board.doMove(null);
215                                            } catch(IllegalMoveException e) {
216                                                    throw new InvalidGameFileException("One player made two moves in a row");
217                                            }
218                                    }
219                                    try {
220                                            board.doMove(m);
221                                    } catch(IllegalMoveException e) {
222                                            throw new InvalidGameFileException("Illegal move: " + m);
223                                    }
224                                    String timestamp = getAttribute(node, "time");
225                                    if(timestamp == null)
226                                            timestamp = "";
227                                    if(moveHistory == null)
228                                            moveHistory = new MoveHistory<ChessMove>(m, timestamp);
229                                    else
230                                            moveHistory = moveHistory.addMove(m, timestamp);
231                            }
232                    }
233                    return moveHistory;
234            }

236            /**
237             * @return the Move represented by the given <move> XML node on
238             * the given board
239             */
240            private ChessMove parseMove(Node node, ChessBoard board)
241                    throws InvalidGameFileException {
242                    Piece piece;
243                    ChessMove move;
244                    Player player = parsePlayer(getAttribute(node, "side"));
245                    String value = getAttribute(node, "value");
246                    move = parseMove(value, board);
247                    piece = move.getPiece();
248                    if(piece == null || piece.getPlayer() != player)
249                            throw new InvalidGameFileException("Move " + value + " moves invalid piece: " + piece);
250                    return move;
251            }

253            /**
254             * @return a ChessMove by parsing the string description of a move
255             * (e.g e2-e4) on the given board.
256             * @throws IllegalArgumentException if the move is ill-formed, or
257             * tries to move a piece that doesn't exist
258             */
259            public static ChessMove parseMove(String value, ChessBoard board) {
260                    Piece piece;
261                    int coords[];
262                    if(!value.matches("[a-h][1-8]-[a-h][1-8]"))
263                            throw new IllegalArgumentException("Bad move:" + value);
264                    coords = parseCoords(value.substring(0,2));
265                    piece = board.getPieceAt(coords[0],coords[1]);
266                    coords = parseCoords(value.substring(3));
267                    return board.createMove(piece, coords[0], coords[1]);
268            }

270            /**
271             * @effects Parse a &lt;gameOver&gt node and set the game over
272             * state on this.board if appropriate.
273             */
274            private void parseGameOver(Node node)
275                    throws InvalidGameFileException {
276                    Player winner = parsePlayer(getAttribute(node, "winner"));
277                    String description = getAttribute(node, "description");
278                    int reason;
279                    if(description.equals("piecesLost"))
280                            reason = AntichessBoard.OUTOFPIECES;
281                    else if(description.equals("checkmate"))
282                            reason = ChessBoard.CHECKMATE;
283                    else if(description.equals("stalemate"))
284                            reason = ChessBoard.STALEMATE;
285                    else if(description.equals("timeExpired"))
286                            reason = Board.OUTOFTIME;
287                    else
288                            throw new InvalidGameFileException("Bad game over reason: " + description);
289                    board.endGame(reason, winner);
290            }

292            /**
293             * @return a list of the Pieces on the board, parsed from the
294             * given <pieces> XML element
295             */
296            private static List<Piece> parsePieces(Node pieces)
297                    throws InvalidGameFileException {
298                    List<Piece> pieceList = new ArrayList<Piece>();
299                    NodeList children;
300                    children = pieces.getChildNodes();
301                    for(int i=0;i<children.getLength();i++) {
302                            Node node = children.item(i);
303                            if("square".equals(node.getNodeName())) {
304                                    pieceList.add(parsePiece(node));
305                            }
306                            
307                    }
308                    return pieceList;
309            }

311            /**
312             * @return the Piece represented by a given XML <square> node
313             */
314            private static Piece parsePiece(Node node)
315                    throws InvalidGameFileException {
316                    Player player = parsePlayer(getAttribute(node, "side"));
317                    int[] coords = parseCoords(getAttribute(node, "id"));
318                    PieceType type = parsePieceType(getAttribute(node, "piece"));
319                    return new Piece(player, type, coords[0], coords[1]);
320            }

322            /**
323             * @return the first child of <tt>node</tt> with type
324             * <tt>type</tt>, or <tt>null</tt> if none exists.
325             */
326            private Node getChildOfType(Node node, int type) {
327                    NodeList children = node.getChildNodes();
328                    for(int i = 0;i < children.getLength(); i++) {
329                            Node child = children.item(i);
330                            if(child.getNodeType() == type)
331                                    return child;
332                    }
333                    return null;
334            }

336            /**
337             * @return the value of the named attribute for <tt>node</tt>, or
338             * <tt>null</tt> if it does not exist.
339             */
340            private static String getAttribute(Node node, String name) {
341                    NamedNodeMap map = node.getAttributes();
342                    if(map == null) return null;
343                    Node attr =  map.getNamedItem(name);
344                    if(attr == null) return null;
345                    return attr.getNodeValue();
346            }

348            /**
349             * @return the Player represented by the given string
350             * @throws InvalidGameFileException if the string is neither "white" nor "black"
351             */
352            private static Player parsePlayer(String player)
353                    throws InvalidGameFileException {
354                    if(player.equals("white"))
355                            return Player.WHITE;
356                    if(player.equals("black"))
357                            return Player.BLACK;
358                    throw new InvalidGameFileException("Bad player specification: " + player);
359            }

361            /**
362             * @return a two-element array of [row, column] representing the
363             * coordinates specified in coords
364             * @requires coords matches "[a-g]\\d"
365             */
366            private static int[] parseCoords(String coords) {
367                    int[] c = new int[2];
368                    c[1] = Character.codePointAt(coords, 0)
369                            - Character.codePointAt("a", 0);
370                    c[0] = Integer.parseInt(coords.substring(1)) - 1;
371                    return c;
372            }

374            /**
375             * @return the PieceType for the given named piece
376             */
377            private static PieceType parsePieceType(String name)
378                    throws InvalidGameFileException {
379                    PieceType type = PieceTypeFactory.getNamedType(name);
380                    if(type == null)
381                            throw new InvalidGameFileException("No such piece type: " + name);
382                    return type;
383            }
