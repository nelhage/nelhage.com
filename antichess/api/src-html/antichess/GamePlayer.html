001    package antichess;
002    
003    /**
004     * The GamePlayer interface allows for communication between 
005     * the contoller and objects that generate moves such as the view
006     * or the AI.
007     * 
008     * @author nlharr
009     *
010     */
011    
012    
013    public interface GamePlayer<M extends Move> {
014            /**
015             * Asks the player for a move, if the player has a move the move
016             * is returned, if the player doesn't, null is returned immediately
017             * Once this GamePlayer is able to return a valid move it calls notifyControllerHasMove()
018             * on the controller.  
019             */
020            public M getMove(GameController controller);
021    
022            /**
023             * Informs the player that this game is over, and it will not be
024             * asked for any more moves. This could happen for a number of
025             * reasons, including one player winning, or a human user
026             * cancelling the game.
027             */
028            public void gameEnded();
029            
030            /**
031             * 
032             * @return Player for this
033             */
034            public Player getPlayerType();
035            
036            
037            /**
038             * If a GamePlayer is automatically timed the time is handled by the GameController
039             * if it is not automatically timed, the time for a move needs to be handled by the GamePlayer
040             * 
041             * 
042             * @return if the time for this player is handled automatically by the 
043             * GameController
044             */
045            public boolean automaticallyTimed();
046    
047    }
