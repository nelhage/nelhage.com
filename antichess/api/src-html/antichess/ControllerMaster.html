012     * <p>
013     * The ControllerMaster is the highest level of the controller.

020     * The ControllerMaster also extends ClassLoader so it can be used for 
021     * loading resources.
022     * </p>

024     * The format needed for a Chess descriptor as input for 
025     * startNewGame and load for a GameDescriptor des:
026     *      des.getPlayers().size() == 2

028     *              des.getPlayers().get(1).equals(Player.BLACK)
029     *              
030     * 
031     * 
032     * @author nlharr
033     *                      
034     * @specfield currentGame                               //String describing the current game
035     * @specfield currentBoard                              //the Board that the current game is running on
036     * @specfield currentClock                              //the clock for the currentgame
037     * @specfield currentUI                                 //the UI that is currently active
038     * 
039     * @specfield description                               //GameDescriptor containing information about the type of game and the playes in the game
040     * @specfield state                                             //the states of the currentController
041     *
042     */

084            private void checkRep(){
085                    if (state == null) throw new RuntimeException("state is null");
086                    if (currentGame != "NONE"){
087                            if (!(currentChessGUI != null || currentTextUI != null)) throw new RuntimeException("No UI");
088                    }
089            
090                    if (currentGame == "Antichess"){
091                            if (currentClock == null) throw new RuntimeException("currentClock is null");
092                            if (currentAntichessBoard == null) throw new RuntimeException("game is antichess currentAntichessBoard is null");
093                            if (currentChessGUI == null) throw new RuntimeException("game is antichess currentChessGUI is null");
094                    }else if (currentGame == "TextAntichess"){
095                            if (currentTextUI == null) throw new RuntimeException("text gamecurrentChessGUI is null");
096                    }else if (currentGame == "Chess"){
097                            if (currentClock == null) throw new RuntimeException("currentClock is null");
098                            if (currentChessBoard == null) throw new RuntimeException("chess game currentChessBoard is null");
099                            if (currentChessGUI == null) throw new RuntimeException("chess game currentChessGUI is null");
100                    }

142            /**
143             * Saves the current game to the the file specified by writeFile.
144             * 
145             * if currentGame is "Chess", save does nothing
146             * 
147             * If currentGame is "Antichess" or "TextAntichess":
148             *              Files are saved in the format specified by:
149             *                      http://www.mit.edu/~6.170/assignments/antichess/antichess.xsd
150             * 
151             *              If one of  the players are untimed it saves the it as if
152             *              both the players are untimed. 
153             *
154             */
155            public void save(File writeFile) throws IOException {
156    
157                            if (currentGame.equals("Antichess")||currentGame.equals("TextAntichess")){
158                                    state.pauseGame();
159                                    
160                                    GameWriter writer;
161                                    
162                                    //gets the writer
163                                    try{
164                                            if (!(currentClock.hasPlayer(Player.WHITE))||!(currentClock.hasPlayer(Player.BLACK))){
165                                                    writer = new GameWriter(currentAntichessBoard, null, null);
166                                            }else{
167                                                    writer = new GameWriter(currentAntichessBoard, currentClock.getTimer(Player.WHITE),
168                                                                                                                       currentClock.getTimer(Player.BLACK));
169                                            }
170                                    }catch (Exception ex){
171                                            state.unpauseGame();
172                                            throw new IOException(ex.getMessage());
173                                    
174                                    }
175                                    //attempts to write the file
176                                    try{
177                                            writer.writeGame(writeFile);
178                                    }catch (Exception ex){
179                                            state.unpauseGame();
180                                            throw new IOException(ex.getMessage());
181                                    }
182                                    
183                                    state.unpauseGame();
184                            
185                            }
186                            checkRep();
187            }

194             *

196             *                      http://www.mit.edu/~6.170/assignments/antichess/antichess.xsd

198             *              by descriptor. 
199             *                      @requires descriptor fit the format for a ChessDescriptor
200             *                      @modifies currentBoard, currentClock, state
201             *                      @effects currentBoard and currentClock are loaded from the file
202             *                                       state the state is set to running 
203             *                      @throws InvalidGameFileException if the readFile does not fit the format
204             *                                      IllegalArgumentException if the descriptor does not fit the format
205             *                                              for a chessDescriptor
206             *                                      IOException if the file does not fit the specified format
207             *
208             * For TextAntichess the player types (human or AI) for the new game on the player types 
209             * of the players previous game.  If there was no previous game then they are both human 
210             * controlled.
211             * 
212             * For Antichess the player types are specified in the descriptor.
213             *
214             * If currentGame is "Chess" the game as an Antichess game and the above rules apply
215             *
216             */

285                            }catch (Exception exc){

287                                    throw new InvalidGameFileException(exc.getMessage());
288                            }
289                            
290                            currentAntichessBoard = reader.getBoard();
291                            GameClock clock;
292                            //initializes the clock
293                            if (reader.isTimed()){
294                                    Player[] players = {Player.WHITE, Player.BLACK};
295                                    long[] startTimes = {reader.getWhiteTimer().getStartTime(),
296                                                                             reader.getBlackTimer().getStartTime()};

305                            }
306

316                                    tempPlayerList.add(Player.WHITE);

530            /**
531             * A private method for creating players
532             * 
533             * @return a GamePlayer of the given named type, for the given board and board view, playing on
534             * the named side using the given clock
535             */
536            private GamePlayer<ChessMove> createPlayer(String type, ChessBoard board, BoardView view, Player side, GameClock clock) {
537                    if (type.equals("Human")){
538                            return new ChessPlayer(board, view, side);
539                    }else if (type.equals("AI")){
540                            if(board instanceof AntichessBoard)
541                                    return new AIPlayer<ChessMove,AntichessBoard>((AntichessBoard)board, side, new AntichessAI());
542                                    //return RandomAI.createPlayer(board, side);
543                            else
544                                    return RandomAI.createPlayer(board, side);
545                    }else if (type.equals("TextAI")){
546                            if(board instanceof AntichessBoard){
547                                    return new TextAIPlayer((AntichessBoard)board, clock, side, side.equals(Player.WHITE)? Player.BLACK : Player.WHITE);
548                            }else{
549                                    throw new RuntimeException("TextAI only for Antichess");
550                            }
551                    }else if (type.equals("TextHuman")){
552                            if(board instanceof AntichessBoard){
553                                    return new TextPlayer((AntichessBoard)board, clock, side);
554                            }else{
555                                    throw new RuntimeException("TextAI only for Antichess");
556                            }
557                    }
558                    
559                    else{
560                            throw new IllegalArgumentException("Improper player description for Chess");
561                    }
562            }

574            /**
575             * @return a GameClock of the type described in the GameDescriptor
576             */
577            public GameClock getClock(GameDescriptor newGame){
578                    //initializes the clock
579                    
580                    //this dense little block filters out all the times that have 0 as there start
581                    //value so that those players games count as untimed
582                    ArrayList<Player> playersClock = new ArrayList<Player>();
583                    ArrayList<Long> timesClock = new ArrayList<Long>();
584                    for (int i=0; i<newGame.getPlayerTimes().length; i++){
585                            if (newGame.getPlayerTimes()[i]!=0){
586                                    playersClock.add(newGame.getPlayers().get(i));
587                                    timesClock.add(Long.valueOf(newGame.getPlayerTimes()[i]));
588                            }
589                    }
590                    //adds the times to an array
591                    long[] timesClockProper = new long[timesClock.size()];
592                    for (int i=0; i<timesClock.size(); i++){
593                            timesClockProper[i]=timesClock.get(i);
594                    }
595                    currentClock = new GameClock(playersClock.toArray(new Player[playersClock.size()]), timesClockProper);
596                    return currentClock;
597            }

599            /**
600            * Confirms that the format of the GameDescriptor newGame
601            * fits for Chess or Antichess. 
602            * The format needed for a Chess descriptor as input for 
603            * startNewGame and load for a GameDescriptor des:
604            *               des.getPlayers().size() == 2
605            *               des.getPlayers().get(0).equals(Player.WHITE)  //checks for order
606            *               des.getPlayers().get(1).equals(Player.BLACK)
607            * @throw exception if the format does not fit 
608            * 
609            */
610            private void checkChessDescriptor(GameDescriptor newGame){
611                    if (newGame == null){
612                            throw new IllegalArgumentException("newgame is null");
613                            
614                    }
615                    //we make sure the GameDescriptor has a proper format for Chess or Antichess
616                    if ((newGame.getPlayers().size()!=2)||
617                                    (newGame.getPlayerDescriptions().size() != 2)||
618                                    (newGame.getPlayerTimes().length != 2)){
619                            throw new IllegalArgumentException("Input Sizes for Chess are wrong");
620                    }
621                    
622                    if ((!newGame.getPlayers().get(0).equals(Player.WHITE))||
623                                    (!newGame.getPlayers().get(1).equals(Player.BLACK))){
624                            throw new IllegalArgumentException("Player order for Chess is wrong");
625                    }
626            }
