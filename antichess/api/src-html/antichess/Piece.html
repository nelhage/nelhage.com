002    
003    package antichess;
004    
005    /**
006     * <p>A <tt>Piece</tt> represents a piece on a board.</p>
007     * <p><tt>Piece</tt>s are immutable</p>
008     *
009     * @specfield player : Player                           // The Player who owns this Piece
010     * @specfield type : PieceType                          // This Piece's type
011     * @specfield position                                          // This Piece's position on the board
012     **/
013    public class Piece
014    {
015            // Abstraction Function:
016            // For all Pieces p:
017            //      player(p) => p.player;
018            //      type(p) => p.type;
019            //      position(p) => (p.row, p.column)
020            
021            private final Player player;
022            private final PieceType type;
023            private final int row;
024            private final int column;
025            private final boolean hasMoved;
026    
027            /**
028             * @effects Constructs a Piece with the given parameters
029             * @requires player != null && type != null
030             */
031            public Piece(Player player, PieceType type, int row, int column) {
032                    this(player, type, row, column, false);
033            }
034            
035            /**
036             * @effects Constructs a Piece with the given parameters
037             * @requires player != null && type != null
038             */
039            private Piece(Player player, PieceType type, int row, int column, boolean hasMoved) {
040                    this.player = player;
041                    this.type = type;
042                    this.row = row;
043                    this.column = column;
044                    this.hasMoved = hasMoved;
045            }
046            
047            /**
048             * @return this.type
049             **/
050            public PieceType getType() {
051                    return type;
052            }
053    
054            /**
055             * @return this.position.row
056             **/
057            public int getRow() {
058                    return row;
059            }
060    
061            /**
062             * @return this.position.column
063             **/
064            public int getColumn() {
065                    return column;
066            }
067    
068    
069            /**
070             * @return this.player
071             **/
072            public Player getPlayer() {
073                    return player;
074            }
075    
076            /**
077             * @return true iff this piece represents a piece that moved
078             **/
079            public boolean hasMoved() {
080                    return hasMoved;
081            }
082            
083            /**
084             * @return a copy of this Piece, relocated to the specified position.
085             **/
086            public Piece moveTo(int row, int column) {
087                    return new Piece(player, type, row, column, true);
088            }
089    
090            /**
091             * @return true if this piece has the same type, player, and
092             * position as the given piece.
093             **/
094            public boolean equals(Object o) {
095                    if(o == null) return false;
096                    if(!(o instanceof Piece)) return false;
097                    Piece p = (Piece)o;
098                    return p.player == player && p.type.equals(type) && p.row == row && p.column == column && p.hasMoved == hasMoved;
099            }
100    
101            @Override
102            public int hashCode() {
103                    int hash = player.hashCode() * 7
104                            + row * 11
105                            + column;
106                    if(hasMoved) hash *= 5;
107                    return hash;
108            }
109    
110            /**
111             * @return a textual description of this piece
112             **/
113            public String toString() {
114                    StringBuilder sb = new StringBuilder();
115                    sb.append("Piece[");
116                    sb.append(player);
117                    sb.append(" ");
118                    sb.append(type.getName());
119                    sb.append("@");
120                    sb.append(row);
121                    sb.append(",");
122                    sb.append(column);
123                    sb.append("]");
124                    return sb.toString();
125            }
126    }
