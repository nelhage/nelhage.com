001    package antichess;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    /**
007     * A GameDescriptor is an immutable ADT that stores the information required
008     * for generating a new game.
009     * 
010     * Fields store by the GameDescriptor:
011     * @specfield gameTypeName              //String which describes the type of game played
012     * @specfield players                           //List<Player> which lists all the Players playing the game
013     * @specfield playerDescriptions        //List<String> which describes the type of player for each
014     * @specfield playerTimes                       //long[] which contains the total amount of playing time for each player
015     *                                                                      //0 means the player is not timed.
016     * @specfield allowSaveLoad                     //boolean as to whether saving and loading are allowed
017     * 
018     * 
019     * Acceptable values for the fields (space delineated)
020     * For gameTypeName             : Chess Antichess TextAntichess
021     * For players                          : WHITE BLACK
022     * For playerDescription        : Human AI TextHuman TextAI
023     * For playerTimes                      : all values must be >= 0
024     * For allowSaveLoad            : true false
025     *
026     * Attempting to create a GameDescriptor with something other than these values
027     * will cause it to throw an IllegalArgumentException
028     * 
029     * Values indexed by the same number in players, playerDescriptions, and playerTimes  
030     * describe the same player.  All of these sequences must be the same size.
031     *
032     * The times stored in playerTimes represent the amount on the timer in milliseconds.
033     *
034     */
035    
036    
037    public class GameDescriptor {
038            
039            
040            //Fields
041            private final String gameTypeName;
042            private final String[] acceptableGameTypeNames = { "Chess", "Antichess", "TextAntichess" };
043            
044            private final List<Player> players;
045            private final Player[] acceptablePlayers = { Player.WHITE, Player.BLACK };
046            
047            private final List<String> playerDescriptions;
048            private final String[] acceptablePlayerDescriptions = {"Human", "AI", "TextHuman", "TextAI"};
049            
050            private final long[] playerTimes;
051            
052            private final boolean allowSaveLoad;
053            
054            //AbstractionFunction
055            //Map the name to the name
056            
057            //Representation Invariants
058            //Encompassed by the actual specs
059            //fields beginning in 'acceptable' store the acceptable values from the spec
060            
061            /**
062             * Constructor 
063             * @requires all parameters to be not null
064             * @modifies all the spec fields
065             * @effects sets the specfield to the argument of the same name
066             * @throws IllegalArgumentException if one of the fields does not contain
067             *      a valid value (For valid values see description of the class)
068             */
069            public GameDescriptor(String gameTypeName,
070                            List<Player> players, List<String> playerDescriptions,
071                            long[] playerTimes, boolean allowSaveLoad){
072                    this.gameTypeName = gameTypeName;
073                    this.players = players;
074                    this.playerDescriptions = playerDescriptions;
075                    this.playerTimes = playerTimes;
076                    this.allowSaveLoad = allowSaveLoad;
077                    checkFields();
078                    
079            }
080            
081            
082            private void checkFields(){
083                    if (gameTypeName == null){
084                            throw new IllegalArgumentException("gameTypeName is null");
085                    }if (players == null){
086                            throw new IllegalArgumentException("players is null");
087                    }if (playerDescriptions == null){
088                            throw new IllegalArgumentException("playerDescriptions is null");
089                    }if (playerTimes == null){
090                            throw new IllegalArgumentException("playerTimes is null");
091                    }
092                    
093                    
094                    if ((players.size()!=playerDescriptions.size())||
095                    (players.size()!=playerTimes.length)){
096                            throw new IllegalArgumentException("Player lists are not the same length");
097                    }
098                    
099                    
100                    //temp variable used to check things
101                    boolean violatesRules;
102                    
103                    //checks for gameTypeName
104                    violatesRules = true;
105                    for (String an : acceptableGameTypeNames){
106                            if (an.equals(gameTypeName)){
107                                    violatesRules = false;
108                            }
109                    }if(violatesRules){
110                            throw new IllegalArgumentException("Invalid gameTypeName "+gameTypeName);
111                    }
112                    
113                    //checks for players
114                    for (Player pl: players){
115                            violatesRules = true;
116                            for (Player pla: acceptablePlayers){
117                                    if (pl.equals(pla))violatesRules = false;
118                            }if(violatesRules){
119                                    throw new IllegalArgumentException("Invalid player "+pl.toString());
120                            }
121                    }
122                    
123                    //checks for playerDescriptions
124                    for (String pl: playerDescriptions){
125                            violatesRules = true;
126                            for (String pla: acceptablePlayerDescriptions){
127                                    if (pl.equals(pla))violatesRules = false;
128                            }if(violatesRules){
129                                    throw new IllegalArgumentException("Invalid playerDescription "+pl);
130                            }
131                    }
132                    
133                    //checks for playerTimes
134                    for (long time : playerTimes ){
135                            if (time<0){
136                                    throw new IllegalArgumentException("Time in playerTimes is negative");
137                            }
138                    }
139                    
140                    
141            }
142            
143            
144            
145            
146            //Observers
147            /**
148             * @return gameTypeName
149             */
150            public String getGameTypeName(){
151                    return gameTypeName;
152            }
153            
154            /**
155             * @return players
156             */
157            public List<Player> getPlayers(){
158                    List<Player> tempList = new ArrayList<Player>();
159                    tempList.addAll(players);
160                    return tempList;
161            }
162            
163            /**
164             * @return playerDescriptions
165             */
166            public List<String> getPlayerDescriptions(){
167                    List<String> tempList = new ArrayList<String>();
168                    tempList.addAll(playerDescriptions);
169                    return tempList;
170            }
171            
172            /**
173             * @return playerTimes
174             */
175            public long[] getPlayerTimes(){
176                    return playerTimes;
177            }
178            
179            /**
180             * @return allowSaveLoad
181             */
182            public boolean getAllowSaveLoad(){
183                    return allowSaveLoad;
184            }
185            
186    }
