003    package antichess;
004    
005    /**
006     * Player is a simple enumeration representing the two sides in a game
007     * of Chess or Antichess.
008     **/
009    public enum Player
010    {
011        WHITE, BLACK, RED, BLUE, NONE;
012    
013        /**
014         * @return WHITE if this == BLACK, BLACK otherwise
015         */
016        public Player otherPlayer() {
017            return this == WHITE ? BLACK : WHITE;
018        }
019    }
