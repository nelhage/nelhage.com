004    package antichess;
005    
006    /**
007     * <p> A <tt>Move</tt> represents a a single move by one player in a
008     * game on a <tt>Board</tt>. A Move contains at least the piece taking
009     * action in the move; It is expected it will be subclassed to contain
010     * more information about specific types of moves for different games.
011     * </p>
012     *
013     * <p> <tt>Move</tt> is immutable, and it is expected that subclasses
014     * will also be. </p>
015     *
016     * @specfield piece  // The piece taking some action
017     **/
018    public abstract class Move
019    {
020            // Abstraction Function:
021            // Identity
022            
023        final Piece piece;
024    
025        protected Move(Piece piece) {
026            this.piece = piece;
027        }
028    
029        /**
030         * @return this.piece
031         */
032        public Piece getPiece() {
033            return piece;
034        }
035        
036        /**
037         * @return the player that made this Move
038         */
039        public Player getPlayer() {
040            return piece.getPlayer();
041        }
042    }
