081            private void checkRep(){
082                    if (master == null) throw new RuntimeException("Master is null");
083                    if (players == null) throw new RuntimeException("players is null");
084                    if (players.keySet().contains(Player.NONE)) throw new RuntimeException("players contains Player.NONE");
085                    for (Player pl : players.keySet()){
086                            if (players.get(pl) == null) throw new RuntimeException("players maps a value to null");
087                    }
088                    if (isRunning){
089                            if(!players.keySet().contains(Player.WHITE)) throw new RuntimeException("players does not contain WHITE");
090                            if(!players.keySet().contains(Player.BLACK)) throw new RuntimeException("players does not contain BLACK");
091                            if(clock == null) throw new RuntimeException("clock is null");
092                            if(board == null) throw new RuntimeException("board is null");
093                    }
094                    
095                    
096            }

116                    this.master = master;

127            /**
128             * @modifies clock, board, isRunning, players
129             * @effects clock = clock, board = board, players is a sequence containing
130             * white and black, isRunning = true
131             */
132            public synchronized void switchGame(AntichessBoard board, GameClock clock,
133                            TextGamePlayer white, TextGamePlayer black, GameController controller){
134                    this.board = board;
135                    this.clock = clock;
136                    players = new TreeMap<Player, TextGamePlayer>();
137                    players.put(white.getPlayerType(), white);
138                    players.put(black.getPlayerType(), black);
139                    this.isRunning = true;
140                    
141                    //Stores internal state information
142                    this.informedGameOver = false;
143                    currentPlayer = Player.NONE;
144    
145                    this.gameStarted = true;
146    
147                    controller.addObserver(this);
148                    
149                    checkRep();
150            }

203            /**
204             * 
205             * @throws IllegalArgumentException if:
206             *                      whiteTime < 0, blackTime < 0
207             *                      !(whitePlayer.equals("human") || whitePlayer.equals("computer"))
208             *                      !(blackPlayer.equals("human") || blackPlayer.equals("computer"))
209             * 
210             *      
211             * Starts a new text game of Antichess with whitePlayer describing 
212             * the type of player white player will be, 'human' and 'computer' 
213             * are acceptable inputs. The same goes for blackPlayer.
214             * 
215             * whiteTime is the starting time for the white player
216             * blackTime is the starting time for the black player
217             * 
218             * If everything goes fine "New game started" is printed to stdout
219             * 
220             */
221            public void startNewGame(String whitePlayer, long whiteTime, 
222                                                            String blackPlayer, long blackTime)
223                    throws IllegalArgumentException{
224                    
225                    if (whiteTime<0 || blackTime <0){
226                            throw new IllegalArgumentException("Negative time input");
227                    }
228                    gameStarted = false;
229                    isRunning = false;
230                    
231                    //creates the playerList
232                    ArrayList<Player> players = new ArrayList<Player>();
233                    players.add(Player.WHITE);
234                    players.add(Player.BLACK);
235                    
236                    //creates the descriptions list
237                    ArrayList<String> descriptions =  new ArrayList<String>();
238                    
239                    //populates the description list with the proper format
240                    if (whitePlayer.equals("human"))
241                            descriptions.add("TextHuman");
242                    else if (whitePlayer.equals("computer"))
243                            descriptions.add("TextAI");
244                    else
245                            throw new IllegalArgumentException("Improper input for white player");
246                    
247                    if (blackPlayer.equals("human"))
248                            descriptions.add("TextHuman");
249                    else if (blackPlayer.equals("computer"))
250                            descriptions.add("TextAI");
251                    else
252                            throw new IllegalArgumentException("Improper input for white player");
253                    
254                    
255                    long[] playerTimes = {whiteTime, blackTime};
256                    
257                    //creates the newGame object
258                    GameDescriptor newGame = new GameDescriptor("TextAntichess",
259                                    players, descriptions,
260                                    playerTimes, true);
261                    
262                    
263                    //tells the master to start a new game
264                    master.startNewGame(newGame);
265                    
266    //              blocks while the controller is making the new game
267                    blockForStartGame();
268                    
269                    System.out.println("New game started");
270                    
271                    
272                    
273            }

276             * Saves the game to the file specified by filename
277             * and then prints "Game saved" to stdout
278             * @requires isRunning == true
279             *                       master != null
280             *                       
281             * @throws Exception if there is some error in the saving process
282             */
283            public void saveGame(String filename) throws Exception{
284                    master.pauseGame();
285                    try{
286                            File file = new File(filename);
287                            master.save(file);
288                    }catch (Exception ex){
289                            master.unpauseGame();
290                            isRunning = true;
291                            throw new RuntimeException(ex.getMessage());
292                    }
293                    master.unpauseGame();
294                    
295                    blockForStartGame();
296                    
297                    System.out.println("Game saved");
298            }

328             * If this command is called during a human player's turn, 
329             * the command prints Human turn on its own line. If this 
330             * command is called during a machine player's turn, print 
331             * on its own line the next move it believes to be the best. 
332             * The printed move should be in the "standard string format" 
333             * described in the assignment. The time taken to compute 
334             * the move should be subtracted from the computer player's 
335             * game clock. If called repeatedly, this should return the 
336             * same move over and over without further decrementing the 
337             * computer's time remaining.
338             *
339             *
340             * @modifies clock 
341             * @effects if its a computer players turn subtracts the time the
342             * computer player uses to generate the move
343             * 
344             */
345            public void getNextMove(){
346                    checkRep(); 
347                    blockForMoveAsk();
348                    players.get(board.getPlayer()).getPossibleMove();
349            }

351            /**
352             * If it is a human player's turn, the system should print 
353             * Please specify human move on its own line. If it is the 
354             * computer player's turn, and GetNextMove has not yet been 
355             * called on this turn, then the system should print 
356             * First GetNextMove. Otherwise, the system performs the move 
357             * that GetNextMove  would return.
358             * 
359             * @modifies board
360             * @effects if getNextMove has been called and it is a computer's turn
361             * causes the move returned by getNextMove to be performed
362             */
363            public void makeNextMove() {
364                    checkRep();
365                    
366                    //to prevent race conditions with accessing currentPlayer
367                    blockForMoveAsk();
368                    moveDone = false;
369                    if (players.get(board.getPlayer()).doPossibleMove()){
370                            blockForMoveDone();
371                    }
372                    
373            }

375            //blocks until it is asked for a move
376            private synchronized void blockForMoveAsk(){
377                    while (currentPlayer.equals(Player.NONE) && master.isRunning()){
378                            try{
379                                    wait(100);
380                            }catch (Exception ex){}
381                    }
382            }
383            
384    //      blocks while the controller is running and attempting to make the move
385            private synchronized void blockForMoveDone(){
386                    while (!moveDone && master.isRunning()){
387                            try{
388                                    wait(100);
389                            }catch (Exception ex){}
390                    }
391                    currentPlayer = Player.NONE;
392            }
393            
394    //      blocks while the controller is making the new game
395            private void blockForStartGame(){
396    
397                    while (!(gameStarted && (master.isRunning() || board.isGameOver()))){
398                            synchronized(this){
399                                    try{
400                                            wait(3000);
401                                    }catch (InterruptedException ex){}
402                            }
403                    }
404            }

425            /**
426             * Perform the move specified by the move string, 
427             * in the "standard string format". The time 
428             * parameter is specified in milliseconds. This command 
429             * should only be used by a Human Player. If it is used 
430             * during a computer player's turn, nothing will happen 
431             * to the game state and no response should be printed. 
432             * If the move is not legal, the system should print, on 
433             * its own line, Illegal move and not perform the move. 
434             * If the move is legal, the system should perform the move, 
435             * decrement the player's time by the amount given, and print 
436             * the move performed, in proper format, on its own line. 
437             * If the player's time is unlimited, then the time argument 
438             * is ignored (but must still be present).
439             *
440             * If a player does not have the ability to move.  That is to say
441             * a player does not have any legal moves their turn is automatically skipped.
442             *
443             * @modifies clock, board
444             * @effects the time for the currentPlayer is decremented by time
445             *                      if the move is legal it is performed on the board
446             */
447            public void makeMove(String move, long time){
448                            checkRep();
449                            
450                            blockForMoveAsk();
451                            moveDone = false;
452                            if (players.get(board.getPlayer()).doMove(move, time)){
453                                    blockForMoveDone();
454                            }
455                    
456            }

459            //places a 2 letter string representation of a piece in a board 
460            private String[] injectPieceCharacters(Piece piece, String[] board){
461                    
462                    int row = 7-piece.getRow();
463                    int column = piece.getColumn();
464                    
465                    String pieceString = "NoPc";
466                    
467                    if (piece.getType().getName().equals("Pawn")){
468                            if (piece.getPlayer().equals(Player.WHITE))
469                                    pieceString = "WPwn";
470                            else
471                                    pieceString = "BPwn";
472                    }else if (piece.getType().getName().equals("Rook")){
473                            if (piece.getPlayer().equals(Player.WHITE))
474                                    pieceString = "WRuk";
475                            else
476                                    pieceString = "BRuk";
477                    }else if (piece.getType().getName().equals("Knight")){
478                            if (piece.getPlayer().equals(Player.WHITE))
479                                    pieceString = "WKgt";
480                            else
481                                    pieceString = "BKgt";
482                    }else if (piece.getType().getName().equals("Bishop")){
483                            if (piece.getPlayer().equals(Player.WHITE))
484                                    pieceString = "WBis";
485                            else
486                                    pieceString = "BBis";
487                    }else if (piece.getType().getName().equals("King")){
488                            if (piece.getPlayer().equals(Player.WHITE))
489                                    pieceString = "WKng";
490                            else
491                                    pieceString = "BKng";
492                    }else if (piece.getType().getName().equals("Queen")){
493                            if (piece.getPlayer().equals(Player.WHITE))
494                                    pieceString = "WQen";
495                            else
496                                    pieceString = "BQen";
497                    }
498                    
499                    
500                    board[4*row + 2] = board[4*row + 2].substring(0, 5+7*column)+pieceString+board[4*row + 2].substring(9+7*column);
501                    
502                    return board;
503            }

579            /**
580             * Prints an alphabetized list of legal moves to stdout
581             *
582             */
583            public void printAllMoves(){
584                    List<ChessMove> moves = board.allLegalMoves();
585                    String[] moveStrings = new String[moves.size()];
586                    //converts all the moves to strings
587                    int i = 0;
588                    for (ChessMove mv : moves){
589                            moveStrings[i] = mv.humanReadableString();
590                            i++;
591                    }
592                    //alphabetizes them
593                    Collator collator = Collator.getInstance(Locale.US);
594                    Arrays.sort(moveStrings, collator );
595                    
596                    //prints the moves out
597                    for (String s : moveStrings){
598                            System.out.println(s);
599                    }
600                    
601            }

743            public void parseInput(String input){
744                    String[] splitInput = input.split(" ");
745                    
746                    if (splitInput.length == 0){
747                            System.out.println("Input error");
748                            return;
749                    }
750                    
751                    if (splitInput[0].equals("StartNewGame")){
752                            if (splitInput.length != 5){
753                                    System.out.println("Input error");
754                                    return;
755                            }
756                            else{
757                                    long time1;
758                                    long time2;
759                                    try{
760                                            time1 = Long.valueOf(splitInput[2]);
761                                            time2 = Long.valueOf(splitInput[4]);
762                                            startNewGame(splitInput[1], time1, splitInput[3], time2);
763                                    }catch (Exception ex){
764                                            System.out.println("Input error");
765                                            return;
766                                    }
767                                    allowQuit = true;
768                                    
769                            }
770                    }else if (splitInput[0].equals("SaveGame")){
771                            if (isRunning){
772                                    if (splitInput.length != 2){
773                                            System.out.println("Input error");
774                                            return;
775                                    }
776                                    try{
777                                            saveGame(splitInput[1]);
778                                    }catch (Exception ex){
779                                            System.out.println("Input error");
780                                            return;
781                                    }
782                            }else{
783                                    System.out.println("Input error");
784                            }
785                    }else if (splitInput[0].equals("LoadGame")){
786                            if (splitInput.length != 2){
787                                    System.out.println("Input error");
788                                    return;
789                            }
790                            try{
791                                    loadGame(splitInput[1]);
792                            }catch (Exception ex){
793                                    System.out.println("Corrupt file");
794                                    return;
795                            }
796                            allowQuit = true;
797                    }else if (splitInput[0].equals("GetNextMove")){
798                            if (isRunning){
799                                    if (splitInput.length == 1){
800                                            getNextMove();
801                                    }else{
802                                            System.out.println("Input error");
803                                            return;
804                                    }
805                            }else{
806                                    System.out.println("Input error");
807                                    return;
808                            }
809                    }else if (splitInput[0].equals("MakeNextMove")){
810                            if (isRunning){
811                                    if (splitInput.length == 1){
812                                            try{
813                                                    makeNextMove();
814                                            }catch (Exception ex){
815                                                    System.out.println("Input error");
816                                                    return;
817                                            }
818                                    }else{
819                                            System.out.println("Input error");
820                                            return;
821                                    }
822                            }else{
823                                    System.out.println("Input error");
824                                    return;
825                            }
826                    }else if (splitInput[0].equals("MakeMove")){
827                            if (isRunning){
828                                    if (splitInput.length == 3){
829                                            long time;
830                                            try{
831                                                    time = Long.valueOf(splitInput[2]);
832                                                    if (time<0) throw new RuntimeException("Input error");
833                                            }catch (Exception ex){
834                                                    System.out.println("Input error");
835                                                    return;
836                                            }
837                                            makeMove(splitInput[1], time);
838                                            
839                                    }else{
840                                            System.out.println("Input error");
841                                            return;
842                                    }
843                            }else{
844                                    System.out.println("Input error");
845                                    return;
846                            }
847                    }else if (splitInput[0].equals("ShowBoard") && this.allowShowBoard){
848                            if (isRunning){
849                                    if (splitInput.length == 1){
850                                            showBoard();
851                                    }else{
852                                            System.out.println("Input error");
853                                            return;
854                                    }
855                            }else{
856                                    System.out.println("Input error");
857                                    return;
858                            }
859                    }else if (splitInput[0].equals("IsLegalMove")){
860                            if (isRunning){
861                                    if (splitInput.length == 2){
862                                            if (isLegalMove(splitInput[1]))
863                                                    System.out.println("legal");
864                                            else
865                                                    System.out.println("illegal");
866                                    }else{
867                                            System.out.println("Input error");
868                                            return;
869                                    }
870                            }else{
871                                    System.out.println("Input error");
872                                    return;
873                            }
874                    }else if (splitInput[0].equals("PrintAllMoves")){
875                            if (isRunning){
876                                    if (splitInput.length == 1){
877                                            printAllMoves();
878

880                                            System.out.println("Input error");
881                                            return;
882                                    }
883                            }else{
884                                    System.out.println("Input error");
885                                    return;
886                            }
887                    }else if (splitInput[0].equals("GetTime")){
888                            if (isRunning){
889                                    if (splitInput.length == 2){
890                                            if (splitInput[1].equals("white")){
891                                                    getTime(Player.WHITE);
892                                            }else if (splitInput[1].equals("black")){
893                                                    getTime(Player.BLACK);
894                                            }else{
895                                                    System.out.println("Input error");
896                                                    return;
897                                            }
898                                            
899                                    }else{

901                                            return;
902                                    }
903                            }else{
904                                    System.out.println("Input error");
905                                    return;
906                            }
907                    }else if (splitInput[0].equals("QuitGame") && allowQuit){
908                            quitGame();
909                            
910                    }else if (splitInput[0].equals("PrintBoard")){
911                            if (isRunning){
912                                    if (splitInput.length == 1){
913                                            printBoard();
914                                    }else{
915                                            System.out.println("Input error");
916                                            return;
917                                    }
918                            }else{
919                                    System.out.println("Input error");
920                                    return;
921                            }
922                    }else{
923                            System.out.println("Input error");
924                            return;
925                    }
