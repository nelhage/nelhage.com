001    package antichess;
002    
003    
004    /**
005    * The TextGamePlayer interface supports methods that the TextUI
006    * will call in order to fit the specifications for the TextUI.
007    * 
008    * 
009    * 
010    *       TextGamePlayer                          
011    *     method                                    TextUI Command
012    *   ------------------------------------------
013    *       doMove                          |               MakeMove
014    *       getPossibleMove         |               GetNextMove
015    *       doPossibleMove          |               MakeNextMove
016    *   ------------------------------------------
017    *   
018    *   @specfield controller               //GameController the controller requesting a move
019    *   @specfield currentMove              //ChessMove that the controller can request with getMove
020    *   @specfield possibleMove             //ChessMove that is cached.  the controller has no access to this.
021    */
022    public interface TextGamePlayer extends GamePlayer<ChessMove> {
023            /**
024             * 
025             * Asks the TextGamePlayer to do a move described by moveString
026             * that takes time milliseconds.  If this TextGamePlayer is a 'human' player
027             * ie TextPlayer then it will set currentMove to the move described
028             * by move string, if the move is illegal it will output "Illegal move".  
029             * 
030             * @modifies currentMove
031             * @effects if is TextPlayer currentMove = the move represented by moveString
032             *                      if is TextAIPlayer no effects
033             * @return true iff the move is legal
034             * 
035             */
036            
037            public boolean doMove(String moveString, long time);
038            
039            
040            /**
041             * Asks the TextGamePlayer to set a possible move if it can.
042             * A TextPlayer is unable to set a move
043             * A TextAIPlayer is able to
044             * 
045             */
046            public void getPossibleMove();
047            
048            
049            /**
050             * Asks the TextGamePlayer to do the move returned by getPossibleMove
051             * If not move is returned by getPossibleMove no move is performed
052             *
053             */
054            public boolean doPossibleMove();
055            
056            
057            
058            
059            
060    }
