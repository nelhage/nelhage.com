001    package antichess;
002    
003    import java.util.*;
004    
005    public class RandomAI implements GamePlayer<ChessMove> {
006    
007            private ChessBoard cb;
008            private Player player;
009            private Random r;
010            
011            /**
012             * Creates a new RandomAI GamePlayer object
013             * 
014             * @param cb
015             * @param player
016             */
017            private RandomAI(ChessBoard cb, Player player) {
018                    this.cb = cb;
019                    this.player = player;
020                    this.r = new Random();
021            }
022            
023            /* (non-Javadoc)
024             * @see antichess.GamePlayer#createPlayer(antichess.Board, antichess.Player)
025             */
026            public static GamePlayer<ChessMove> createPlayer(ChessBoard board, Player player) {
027                    return new RandomAI(board, player);
028            }
029    
030            /* (non-Javadoc)
031             * @see antichess.GamePlayer#getMove()
032             */
033            public ChessMove getMove(GameController controller) {
034                    /*try{
035                            Thread.sleep(100);
036                    }catch (Exception ex){}
037                    */
038                    
039                    List<ChessMove> legalMoves = cb.allLegalMoves();
040                    
041                    return legalMoves.get(r.nextInt(legalMoves.size()));
042            }
043    
044            public void gameEnded() {}
045    
046             
047            
048            public Player getPlayerType(){
049                    return this.player;
050            }
051            
052            public boolean automaticallyTimed(){
053                    return true;
054            }
055            
056            
057    }
