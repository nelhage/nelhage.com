003    package antichess.movegen;
004    
005    import antichess.*;
006    import java.util.*;
007    
008    /**
009     * A UnionMoveGenerator is a move generator that unions the moves
010     * produced by two MoveGenerators
011     * 
012     **/
013    public class UnionMoveGenerator<M extends Move> implements MoveGenerator<M>
014    {
015        private MoveGenerator<M> move1;
016            private MoveGenerator<M> move2;
017    
018            /**
019             * @effects Construct a MoveGenerator that returns the union of
020             * the moves of the two generators
021             */
022            public UnionMoveGenerator(MoveGenerator<M> m1, MoveGenerator<M> m2) {
023                    move1 = m1;
024                    move2 = m2;
025            }
026    
027            public List<M> getMoves(Piece piece, Board<M> board) {
028                    List<M> moves = move1.getMoves(piece, board);
029                    moves.addAll(move2.getMoves(piece, board));
030                    return moves;
031            }
032    
033            public boolean isMoveLegal(M move, Board<M> board) {
034                    return move1.isMoveLegal(move, board)
035                            || move2.isMoveLegal(move, board);
036            }
037    
038    }
