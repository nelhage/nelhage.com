003    package antichess;
004    
005    /**
006     * A <tt>ChessMove</tt> represents a move in a Chess game.
007     * <tt>ChessMove</tt>s are immutable.
008     *
009     * @specfield row : int                    // The row the piece is moving to
010     * @specfield column : int                 // The column the piece is moving to 
011     * @specfield capturedPiece : Piece     // The piece that is captured by this Move. May be null.
012     **/
013    public class ChessMove extends Move
014    {
015            // Abstraction Function:
016            // Identity.
017            
018            // Representation Invariant:
019            // For all ChessMoves cm:
020            //      cm.capturedPiece == null or a valid Piece.
021            
022            private int row;
023            private int column;
024            // Optional piece that is caputred. Stored in its state before capture.
025            private Piece capturedPiece;
026            
027            /**
028             * @param piece the piece for this ChessMove
029             * @param row the row to which that piece is moving
030             * @param column the column to which that piece is moving
031             * @param capturedPiece the optional captured piece
032             * @effects Construct a new ChessMove with the given values.
033             */
034            public ChessMove(Piece piece, int row, int column, Piece capturedPiece) {
035                    super(piece);
036                    this.row = row;
037                    this.column = column;
038                    this.capturedPiece = capturedPiece;
039            }
040            
041            /**
042             * Helper constructor that assumes the capturedPiece is null
043             * @effects Construct a new ChessMove with the given values.
044             * @param piece the piece for this ChessMove
045             * @param row the row to which that piece is moving
046             * @param column the column to which that piece is moving
047             */
048            public ChessMove(Piece piece, int row, int column) {
049                    this(piece, row, column, null);
050            }
051    
052            /**
053             * @return this.row
054             */
055            public int getRow() {
056                    return row;
057            }
058    
059            /**
060             * @return this.column
061             */
062            public int getColumn() {
063                    return column;
064            }
065            
066            /**
067             * @return this.capturedPiece
068             */
069            public Piece getCapturedPiece() {
070                    return capturedPiece;
071            }
072    
073            /**
074             * @return a textual description of this move
075             **/
076            public String toString() {
077                    StringBuilder sb = new StringBuilder();
078                    sb.append("ChessMove[");
079                    sb.append(piece);
080                    sb.append(" -> ");
081                    sb.append(row);
082                    sb.append(",");
083                    sb.append(column);
084                    if (capturedPiece != null) {
085                            sb.append(" takes ");
086                            sb.append(capturedPiece);
087                    }
088                    sb.append("]");
089                    return sb.toString();
090            }
091            
092            /**
093             * @param row
094             * @return the standard chess representation of a row
095             */
096            private String chessRow(int row) {
097                    return Integer.toString(row+1);
098            }
099            
100            /**
101             * @param column
102             * @return the standard chess representation of a column
103             */
104            private String chessColumn(int column) {
105                    return Integer.toString(column+10, 18);
106            }
107            
108            /**
109             * @return a textual representation of the move
110             * that makes sense to a user
111             */
112            // And by "makes sense to a user" I mean "conforms
113            // to their retarded specs."
114            public String humanReadableString() {
115                    StringBuilder sb = new StringBuilder();
116                    int startRow = piece.getRow(), startCol = piece.getColumn();
117                    sb.append(chessColumn(startCol));
118                    sb.append(chessRow(startRow));
119                    sb.append("-");
120                    sb.append(chessColumn(column));
121                    sb.append(chessRow(row));
122                    return sb.toString();
123            }
124            
125            /**
126             * @param o the Object against which to test equality
127             * @return true if the Object o is a ChessMove equal to this; false otherwise
128             */
129            @Override
130            public boolean equals(Object o) {
131                    if (!(o instanceof ChessMove))
132                            return false;
133                    ChessMove cm = (ChessMove)o;
134                    if ((this.row == cm.row) && (this.column == cm.column)) {
135                            if ((this.capturedPiece == null && this.capturedPiece == cm.capturedPiece) ||
136                                    (this.capturedPiece != null && this.capturedPiece.equals(cm.capturedPiece)))
137                                    return true;
138                            else
139                                    return false;
140                    }
141                    return false;
142            }
143    
144            public int hashCode() {
145                    int hashCode = piece.hashCode() * 5
146                            + row * 7
147                            + column * 3;
148                    if(capturedPiece != null)
149                            hashCode += capturedPiece.hashCode() * 17;
150                    return hashCode;
151            }
152    }
