003    package antichess.ai;
004    
005    import antichess.*;
006    
007    /**
008     * <p>The <code>GameAI</code> class represents an abstract AI over a
009     * <code>Board</code>.</p>
010     **/
011    public interface GameAI<M extends Move, B extends Board<M> >
012    {
013            /**
014             * <p>
015             * Return a move computed by the AI for the given board, with the
016             * given amount of time remaining, in milliseconds.
017             * </p>
018             * <p>
019             * A time of 0 for either time means that player is untimed.
020             * </p>
021             * @param timeLeft : The time left on the current player's clock
022             * @param opponentTimeLeft : The time remaining on the other player's clock
023             */
024            public abstract M findMove(B board, long timeLeft, long opponentTimeLeft);
025    }
