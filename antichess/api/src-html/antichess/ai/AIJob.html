003    package antichess.ai;
004    
005    import antichess.*;
006    
007    /**
008     * An <code>AIJob</code> is a game tree node to be evaluated by a
009     * <code>MinimaxAI.AIThread</code>
010     * 
011     * @specfield board : B         // The Board to be evaluated
012     * @specfield move : M          // The Move to evaluate on the board
013     * @specfield depth     : int   // The depth to which to evaluate the board
014     * @specfield bestMove : AIMove<M>        // The best move found so far from the <em>parent</em>
015            * of the board in the game tree
016     * @specfield beta : int        // The value of beta for alpha-beta pruning the search
017     * @specfield tryMove : AIMove<M> // A line of moves that the caller has reason to believe
018             * represents a good move from <code>board</code> after making
019             * <code>move</code>
020     **/
021    public class AIJob<M extends Move, B extends Board<M>>
022    {
023            /**
024             * The board to be evaluated
025             */
026            public B board;
027    
028            /**
029             * The move to evaluate on the board
030             */
031            public M move;
032    
033            /**
034             * The depth to which to evaluate the board
035             */
036            int depth;
037    
038            /**
039             * The best move found so far for the <em>parent</em> of the board
040             * in the game tree. This will be updated by threads as they
041             * complete their search.
042             */
043            public AIMove<M> bestMove;
044    
045            /**
046             * The value of beta for alpha-beta pruning the search
047             */
048            public int beta;
049            
050            /**
051             * A line of moves that the caller has reason to believe
052             * represents a good move from <code>board</code> after making
053             * <code>move</code>. This may be <code>null</code>. Specifying a
054             * move here does not change the semantics of the method, but
055             * specifying a good move may potentially speed up the search. It
056             * is not required that <code>tryMove.move</code> be a legal move
057             * from the given board.
058             */
059            public AIMove<M> tryMove;
060    }
