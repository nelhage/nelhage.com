003    package antichess.ai;
004    
005    import antichess.*;
006    
007    /**
008     * A <code>BoardEvaluator</code> contains code to generate a static
009     * evaluation of a <code>Board</code>
010     **/
011    public interface BoardEvaluator<B extends Board<? extends Move>>
012    {
013            /**
014             * @return a heuristic value for the given board. Larger values
015             * should indicate positions more favorable for the White player.
016             */
017        public int boardValue(B board);
018    }
